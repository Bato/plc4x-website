MSpec Code-Generation Documentation

Simple types

Some times no matter what content is coming, the same type is used. In this case a type is used:

[type 'SimpleType'
    [array         uint 8 'arrCount'      count      '5']
    [array         uint 8 'arrLen'        length     '6']
    [array         uint 8 'arrTerm'       terminated                'terminationExpression']
    [checksum      uint 8 'crc'           'checksumExpression']
    [const         uint 8 'con'           '0x03']
    [implicit      uint 8 'impl'          'serializationExpression']
    [manualArray   uint 8 'manArrayCount' count                     'loopExpression'            'serializationExpression' 'deserializationExpression' 'lengthExpression']
    [manualArray   uint 8 'manArrayLen'   length                    'loopExpression'            'serializationExpression' 'deserializationExpression' 'lengthExpression']
    [manualArray   uint 8 'manArrayTerm'  terminated                'loopExpression'            'serializationExpression' 'deserializationExpression' 'lengthExpression']
    [manual        uint 8 'man'           'serializationExpression' 'deserializationExpression' 'lengthExpression']
    [optional      uint 8 'opt'           'optionalExpression']
    [padding       uint 8 'pad'           '0'                       'paddingExpression']
    [reserved      uint 8 '0x00']
    [simple        uint 8 'simp']
    [virtual       uint 8 'virt'          'valueExpression']
]

The code generator will create simple POJOs as well as serializers and parsers from this specification.

Discriminated types

There are however cases, where the content of the packet determines itâs type. In these case a discriminatedType must be used. Each discriminatedType must have one discriminator field and one typeSwitch field to define the sub-types.

[discriminatedType 'DiscriminatedType'
    [discriminator uint 8 'discr']
    [typeSwitch 'discr'
        ['0x01' DiscriminatedTypeA
            [simple        uint 8 'simpA']
        ]
        ['0x02' DiscriminatedTypeA
            [simple        uint 8 'simpB']
        ]
        ['0x03' DiscriminatedTypeA
            [simple        uint 8 'simpC']
        ]
    ]
]

In contrast to the simple types, in this case a base-type POJO will be created, which contains all the fields all sub-types share. However these types are generated as abstract types. The subtypes then extend these base-types.

Field types

In this section comes a description of all the different types of fields supported by the code generator.

array

An array field will generate an array property. There are 3 variants of arrays which differ in the rule for how many elements the array should contain.

Right now these are: - count: An explicit number of element will be read - length: The number of bytes the entire array will consume (The parser will continue reading elements until the given number of bytes are read) - terminated: The parser will continue to read elements until a given exit termination condition is true.

    [array         uint 8 'arrCount'      count      '5']
    [array         uint 8 'arrLen'        length     '6']
    [array         uint 8 'arrTerm'       terminated                'terminationExpression']

The general structure is:

[array {simple-type|complex-type} '{name}' {count|length|terminated} '{expression}']

    public static SimpleType parse(ReadBuffer io) throws ParseException {
        int startPos = io.getPos();
        int curPos;

        // Array field (arrCount)
        // Count array
        int _arrCountCount = 5;
        short[] arrCount = new short[_arrCountCount];
        for(int i = 0; i < _arrCountCount; i++) {
            arrCount[i] = io.readUnsignedShort(8);
        }

        // Array field (arrLen)
        // Length array
        int _arrLenLength = 6;
        List<Short> _arrLenList = new LinkedList<>();
        int arrLenEndPos = io.getPos() + _arrLenLength;
        while(io.getPos() < arrLenEndPos) {
            _arrLenList.add(io.readUnsignedShort(8));
        }
        short[] arrLen = new short[_arrLenList.size()];
        for(int i = 0; i < _arrLenList.size(); i++) {
            arrLen[i] = (short) _arrLenList.get(i);

    public static void serialize(WriteBuffer io, SimpleType value) throws ParseException {

        // Array Field (arrCount)

package org.apache.plc4x.java.test.readwrite;



import com.fasterxml.jackson.annotation.*;
import org.apache.plc4x.java.utils.Message;


public class SimpleType implements Message {


    // Constant values.
    public static final short CON = 0x03;

    // Properties.
    private final short[] arrCount;
    private final short[] arrLen;
    private final short[] arrTerm;
    private final short[] manArrayCount;
    private final short[] manArrayLen;
    private final short[] manArrayTerm;
    private final short man;
    private final Short opt;
    private final short simp;

    @JsonCreator(mode = JsonCreator.Mode.PROPERTIES)
    public SimpleType(@JsonProperty("arrCount") short[] arrCount, @JsonProperty("arrLen") short[] arrLen, @JsonProperty("arrTerm") short[] arrTerm, @JsonProperty("manArrayCount") short[] manArrayCount, @JsonProperty("manArrayLen") short[] manArrayLen, @JsonProperty("manArrayTerm") short[] manArrayTerm, @JsonProperty("man") short man, @JsonProperty("opt") Short opt, @JsonProperty("simp") short simp) {
        this.arrCount = arrCount;
        this.arrLen = arrLen;
        this.arrTerm = arrTerm;
        this.manArrayCount = manArrayCount;
        this.manArrayLen = manArrayLen;
        this.manArrayTerm = manArrayTerm;
        this.man = man;
        this.opt = opt;
        this.simp = simp;
    }

    public short[] getArrCount() {
        return arrCount;
    }

    public short[] getArrLen() {
        return arrLen;
    }

    public short[] getArrTerm() {
        return arrTerm;
    }

    public short[] getManArrayCount() {
        return manArrayCount;
    }

    public short[] getManArrayLen() {
        return manArrayLen;
    }

    public short[] getManArrayTerm() {
        return manArrayTerm;
    }

    public short getMan() {
        return man;
    }

    public Short getOpt() {
        return opt;
    }

    public short getSimp() {
        return simp;
    }

    public short getVirt() {
        return (short) (valueExpression);
    }

    @Override
    @JsonIgnore
    public int getLengthInBytes() {
        int lengthInBits = 0;

        // Array field
        lengthInBits += 8 * arrCount.length;

        // Array field
        lengthInBits += 8 * arrLen.length;

        // Array field
        lengthInBits += 8 * arrTerm.length;

        // Checksum Field (crc)
        lengthInBits += 8;

        // Const Field (con)
        lengthInBits += 8;

        // Implicit Field (impl)
        lengthInBits += 8;

        // Manual Array Field (manArrayCount)
        lengthInBits += lengthExpression * 8;

        // Manual Array Field (manArrayLen)
        lengthInBits += lengthExpression * 8;

        // Manual Array Field (manArrayTerm)
        lengthInBits += lengthExpression * 8;

        // Manual Field (man)
        lengthInBits += lengthExpression * 8;

        // Optional Field (opt)
        if(opt != null) {
            lengthInBits += 8;
        }

        // Padding Field (pad)
        if((boolean) (paddingExpression)) {
            lengthInBits += 8;
        }

        // Reserved Field
        lengthInBits += 8;

        // Simple field (simp)
        lengthInBits += 8;

        // A virtual field doesn't have any in- or output.

        return lengthInBits / 8;
    }

}
