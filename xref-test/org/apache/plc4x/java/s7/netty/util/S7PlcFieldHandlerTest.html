261     private static Stream<Arguments> createInputArrays() {
262         // Generate valid fields for each s7 type.
263         Map<TransportSize, PlcField> fields = new HashMap<>();
264         for (TransportSize s7Type : TransportSize.values()) {
265             String sizeCode = (s7Type.getSizeCode() != null) ? s7Type.getSizeCode() : "X";
266             String fieldQuery = "%DB1.DB" + sizeCode + "1";
267             if(s7Type == TransportSize.BOOL) {
268                 fieldQuery += ".0";
269             }
270             fieldQuery += ":" + s7Type.name();
271             fields.put(s7Type, SUT.createField(fieldQuery));
272         }
273         // Generate output for each combination of S7 and Java type.
274         Stream<Arguments> values = null;
275         for (TransportSize s7Type : TransportSize.values()) {
276             PlcField field = fields.get(s7Type);
277             for (InputTypes javaType : InputTypes.values()) {
278                 Object[] testValues = javaType.values;
279 
280                 BaseDefaultFieldItem fieldItem;
281                 try {
282                     fieldItem = javaType.fieldItemType.getDeclaredConstructor(Array.newInstance(testValues[0].getClass(), 0).getClass()).newInstance(new Object[]{testValues});
283                 } catch (InstantiationException | IllegalAccessException | InvocationTargetException | NoSuchMethodException e) {
284                     throw new PlcRuntimeException("Error initializing field class " + javaType.fieldItemType.getSimpleName(), e);
285                 }
286 
287                 Stream<Arguments> curValues;
288                 // Min, Max
289                 if(testValues.length == 2) {
290                     curValues = Stream.of(
291                         Arguments.of(s7Type.name() + "-" + javaType.name() + "-MIN", field, createOneElementArray(testValues[0])),
292                         Arguments.of(s7Type.name() + "-" + javaType.name() + "-MAX", field, createOneElementArray(testValues[1])));
293                 }
294                 // Value, Min, Max
295                 else if(testValues.length == 3) {
296                     curValues = Stream.of(
297                         Arguments.of(s7Type.name() + "-" + javaType.name() + "-MIN", field, createOneElementArray(testValues[1])),
298                         Arguments.of(s7Type.name() + "-" + javaType.name() + "-VAL", field, createOneElementArray(testValues[0])),
299                         Arguments.of(s7Type.name() + "-" + javaType.name() + "-MAX", field, createOneElementArray(testValues[2])));
300                 }
301                 // Zero, Value, Min, Max
302                 else if(testValues.length == 4) {
303                     curValues = Stream.of(
304                         Arguments.of(s7Type.name() + "-" + javaType.name() + "-MIN", field, createOneElementArray(testValues[2])),
305                         Arguments.of(s7Type.name() + "-" + javaType.name() + "-NIL", field, createOneElementArray(testValues[0])),
306                         Arguments.of(s7Type.name() + "-" + javaType.name() + "-VAL", field, createOneElementArray(testValues[1])),
307                         Arguments.of(s7Type.name() + "-" + javaType.name() + "-MAX", field, createOneElementArray(testValues[3])));
308                 } else {
309                     throw new RuntimeException("Expecting 2, 3 or 4 valued test-input");
310                 }
311                 if(values == null) {
312                     values = curValues;
313                 } else {
314                     values = Stream.concat(values, curValues);
315                 }
316             }
317         }
318         return values;
319     }

321     private static Object[] createOneElementArray(Object value) {
322         Class<?> type = value.getClass();
323         Object[] array = (Object[]) Array.newInstance(type, 1);
324         array[0] = value;
325         return array;
326     }

328     enum InputTypes {
329         BOOLEAN(DefaultBooleanFieldItem.class, new Boolean[]{false, true}),
330         BYTE(DefaultByteFieldItem.class, new Byte[]{(byte) 0, (byte) 42, Byte.MIN_VALUE, Byte.MAX_VALUE}),
331         SHORT(DefaultShortFieldItem.class, new Short[]{(short) 0, (short) 42, Short.MIN_VALUE, Short.MAX_VALUE}),
332         INTEGER(DefaultIntegerFieldItem.class, new Integer[]{0, 42, Integer.MIN_VALUE, Integer.MAX_VALUE}),
333         LONG(DefaultLongFieldItem.class, new Long[]{(long) 0, (long) 42, Long.MIN_VALUE, Long.MAX_VALUE}),
334         FLOAT(DefaultFloatFieldItem.class, new Float[]{(float) 0.0, (float) 42.23, -Float.MAX_VALUE, Float.MAX_VALUE}),
335         DOUBLE(DefaultDoubleFieldItem.class, new Double[]{0.0, 42.23, -Double.MAX_VALUE, Double.MAX_VALUE}),
336         // Creates an empty sting as min and a 254 char long string as max.
337         STRING(DefaultStringFieldItem.class, new String[]{"Hurz", "", IntStream.range(0, 254).mapToObj(i -> "a").collect(Collectors.joining(""))}),
338         TIME(DefaultLocalTimeFieldItem.class, new LocalTime[]{LocalTime.now(), LocalTime.MIDNIGHT, LocalTime.MIN, LocalTime.MAX}),
339         DATE(DefaultLocalDateFieldItem.class, new LocalDate[]{LocalDate.now(), LocalDate.MIN, LocalDate.MAX}),
340         DATETIME(DefaultLocalDateTimeFieldItem.class, new LocalDateTime[]{LocalDateTime.now(), LocalDateTime.MIN, LocalDateTime.MAX});
341 
342         private final Class<? extends BaseDefaultFieldItem> fieldItemType;
343         private final Object[] values;
344 
345         InputTypes(Class<? extends BaseDefaultFieldItem> fieldItemType, Object[] values) {
346             this.fieldItemType = fieldItemType;
347             this.values = values;
348         }
349     }

351     private void encode(String name, PlcField field, Object[] values, Set<String> expectedSuccess,
352                         BiFunction<PlcField, Object[], BaseDefaultFieldItem> encoder) {
353         boolean success = expectedSuccess.contains(name);
354         try {
355             BaseDefaultFieldItem fieldItem = encoder.apply(field, values);
356             assertNotNull(fieldItem, "A FieldItem instance should have been returned for testcase " + name);
357             if(!success) {
358                 fail("Expected to fail for testcase " + name);
359             }
360         } catch (Exception e) {
361             if(success) {
362                 fail("Expected to succeed for testcase " + name + " got exception " + e.getMessage());
363             }
364         }
365     }
