32  import java.util.stream.Stream;
33  
34  import static org.hamcrest.core.IsEqual.equalTo;
35  import static org.hamcrest.core.IsNull.notNullValue;
36  import static org.junit.Assert.assertEquals;
37  import static org.junit.Assert.assertThat;
38  import static org.junit.Assert.fail;
39  
40  class S7FieldTests {
41  
42      private static Stream<Arguments> validFieldQueries() {
43          return Stream.of(
44              Arguments.of("%I0.1:BOOL",          TransportSize.BOOL,  MemoryArea.INPUTS,      0,  0,  1),
45              Arguments.of("%ID64:REAL",          TransportSize.REAL,  MemoryArea.INPUTS,      0,  64, 0),
46              Arguments.of("%Q0.4:BOOL",          TransportSize.BOOL,  MemoryArea.OUTPUTS,     0,  0,  4),
47              Arguments.of("%DB1.DBX38.1:BOOL",   TransportSize.BOOL,  MemoryArea.DATA_BLOCKS, 1,  38, 1)/*,
48              // Not quite sure about how Data Block addresses look like, in my TIA portal they all have the prefix "DB".
49              Arguments.of("%DB3.DX4.1:BOOL",     S7DataType.BOOL,  MemoryArea.DATA_BLOCKS, 3,  4,  1),
50              Arguments.of("%DB3.DB4:INT",        S7DataType.INT,   MemoryArea.DATA_BLOCKS, 3,  4,  0),
51              Arguments.of("%DB3.DB4:UINT",       S7DataType.UINT,  MemoryArea.DATA_BLOCKS, 3,  4,  0),
52              Arguments.of("%DB3.DW4:REAL",       S7DataType.REAL,  MemoryArea.DATA_BLOCKS, 3,  4,  0)*/
53          );
54      }
55  
56      private static Stream<Arguments> invalidFieldQueries() {
57          return Stream.of(
58              Arguments.of("%I0:BOOL"),
59              Arguments.of("%IW64:REAL"),
60              Arguments.of("%DB1.DBX38:BOOL")
61          );
62      }
63  
64      @ParameterizedTest
65      @Category(FastTests.class)
66      @MethodSource("validFieldQueries")
67      void testValidFieldQueryParsing(String fieldQuery, TransportSize expectedClientType, MemoryArea expectedMemoryArea,
68                                      int expectedMemoryBlockNumber, int expectedByteOffset, int expectedBitOffset) {
69          S7Field field = S7Field.of(fieldQuery);
70          assertThat(field, notNullValue());
71          assertThat(field.getDataType(), equalTo(expectedClientType));
72          assertThat(field.getMemoryArea(), equalTo(expectedMemoryArea));
73          assertThat(field.getBlockNumber(), equalTo((short) expectedMemoryBlockNumber));
74          assertThat(field.getByteOffset(), equalTo((short) expectedByteOffset));
75          assertThat(field.getBitOffset(), equalTo((short) expectedBitOffset));
76      }
77  
78      @ParameterizedTest
79      @Category(FastTests.class)
80      @MethodSource("invalidFieldQueries")
81      void testInvalidFieldQueryParsing(String fieldQuery) {
82          try {
83              S7Field.of(fieldQuery);
84              fail("Should have thrown an exception");
85          } catch (PlcRuntimeException e) {
86              // This was expected.
87          }
88      }
89  
90      @Test
91      void checkGreedyNumFieldsParsing() {
92          S7Field field = S7Field.of("%DB56.DBB100:SINT[25]");
93  
94          assertEquals(25, field.getNumElements());
95      }
96  }
