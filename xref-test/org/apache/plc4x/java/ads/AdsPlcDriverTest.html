48      @Test
49      public void testAdsAddressPattern() {
50          assertMatching(ADS_ADDRESS_PATTERN, "0.0.0.0.0.0:13");
51          assertMatching(ADS_ADDRESS_PATTERN, "0.0.0.0.0.0:13/0.0.0.0.0.0:13");
52  
53          assertMatching(INET_ADDRESS_PATTERN, "tcp://localhost");
54          assertMatching(INET_ADDRESS_PATTERN, "tcp://localhost:3131");
55          assertMatching(INET_ADDRESS_PATTERN, "tcp://www.google.de");
56          assertMatching(INET_ADDRESS_PATTERN, "tcp://www.google.de:443");
57  
58          assertMatching(SERIAL_PATTERN, "serial:///dev/com1");
59          assertMatching(SERIAL_PATTERN, "serial://COM1");
60          assertMatching(SERIAL_PATTERN, "serial:///dev/ttyUSB0");
61  
62          assertMatching(ADS_URI_PATTERN, "ads:tcp://www.google.de/0.0.0.0.0.0:13");
63          assertMatching(ADS_URI_PATTERN, "ads:tcp://www.google.de:443/0.0.0.0.0.0:13");
64          assertMatching(ADS_URI_PATTERN, "ads:serial:///dev/com1/0.0.0.0.0.0:13");
65          assertMatching(ADS_URI_PATTERN, "ads:serial://COM1/0.0.0.0.0.0:13");
66          assertMatching(ADS_URI_PATTERN, "ads:serial:///dev/ttyUSB0/0.0.0.0.0.0:13");
67  
68          assertMatching(ADS_URI_PATTERN, "ads:serial:///dev/ttyUSB0/0.0.0.0.0.0:13?some=random&parameters=true");
69      }

71      @Test
72      public void testDriverWithCompleteUrls() {
73          AdsPlcDriver SUT = new AdsPlcDriver(mock(AdsConnectionFactory.class));
74          Stream.of(
75              "ads:tcp://www.google.de/0.0.0.0.0.0:13",
76              "ads:tcp://www.google.de:443/0.0.0.0.0.0:13",
77              "ads:serial:///dev/com1/0.0.0.0.0.0:13",
78              "ads:serial://COM1/0.0.0.0.0.0:13",
79              "ads:serial:///dev/ttyUSB0/0.0.0.0.0.0:13"
80          ).forEach(url -> {
81              try {
82                  SUT.connect(url);
83              } catch (PlcConnectionException e) {
84                  throw new PlcRuntimeException(e);
85              }
86          });
87      }
88  
89      private void assertMatching(Pattern pattern, String match) {
90          if (!pattern.matcher(match).matches()) {
91              fail(pattern + "doesn't match " + match);
92          }
93      }
94  
95      @Test
96      public void getConnection() throws Exception {
97          AdsTcpPlcConnection adsConnection = (AdsTcpPlcConnection)
98              new PlcDriverManager().getConnection("ads:tcp://localhost:" + tcpHexDumper.getPort() + "/0.0.0.0.0.0:13");
99          assertEquals(adsConnection.getTargetAmsNetId().toString(), "0.0.0.0.0.0");
100         assertEquals(adsConnection.getTargetAmsPort().toString(), "13");
101         adsConnection.close();
102     }

110     @Test
111     public void getConnectionUnknownHost() {
112         try {
113             new PlcDriverManager().getConnection("ads:tcp://nowhere:8080/0.0.0.0.0.0:13");
114             fail();
115         } catch (Exception e) {
116             assert((e instanceof PlcConnectionException) ||
117                 // When running this test on systems with an internet provider that redirects
118                 // all unknown hosts to a default host (For showing adds), one of these will
119                 // be thrown instead.
120                 (e instanceof ConnectTimeoutException) ||
121                 (e.getCause() instanceof ConnectException));
122         }
123     }
