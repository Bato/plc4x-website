187     private PlcEntityManager getPlcEntityManager(final Map<String, BaseDefaultFieldItem> responses) throws PlcConnectionException {
188         driverManager = Mockito.mock(PlcDriverManager.class);
189         PlcDriverManager mock = driverManager;
190         PlcConnection connection = Mockito.mock(PlcConnection.class);
191         when(mock.getConnection(ArgumentMatchers.anyString())).thenReturn(connection);
192         when(connection.getMetadata()).thenReturn(new PlcConnectionMetadata() {
193 
194             @Override
195             public boolean canRead() {
196                 return true;
197             }
198 
199             @Override
200             public boolean canWrite() {
201                 return true;
202             }
203 
204             @Override
205             public boolean canSubscribe() {
206                 return true;
207             }
208         });
209 
210         PlcReader reader = readRequest -> {
211             Map<String, Pair<PlcResponseCode, BaseDefaultFieldItem>> map = readRequest.getFieldNames().stream()
212                 .collect(Collectors.toMap(
213                     Function.identity(),
214                     s -> Pair.of(PlcResponseCode.OK, Objects.requireNonNull(responses.get(s), s + " not found"))
215                 ));
216             return CompletableFuture.completedFuture(new DefaultPlcReadResponse((InternalPlcReadRequest) readRequest, map));
217         };
218         when(connection.readRequestBuilder()).then(invocation -> new DefaultPlcReadRequest.Builder(reader, getFieldHandler()));
219 
220         return new PlcEntityManager(mock);
221     }
