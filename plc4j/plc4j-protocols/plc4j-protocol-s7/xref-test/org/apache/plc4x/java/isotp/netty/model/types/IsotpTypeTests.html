26  import static org.hamcrest.core.IsEqual.equalTo;

31  public class IsotpTypeTests {
32  
33      @Test
34      @Category(FastTests.class)
35      public void deviceGroup() {
36          DeviceGroup deviceGroup;
37  
38          deviceGroup = DeviceGroup.PG_OR_PC;
39          assertThat(deviceGroup.getCode(), equalTo((byte) 1));
40  
41          deviceGroup = DeviceGroup.OS;
42          assertThat(deviceGroup.getCode(), equalTo((byte) 2));
43  
44          deviceGroup = DeviceGroup.OTHERS;
45          assertThat(deviceGroup.getCode(), equalTo((byte) 3));
46      }
47  
48      @Test
49      @Category(FastTests.class)
50      public void deviceGroupUnknown() {
51          DeviceGroup deviceGroup = DeviceGroup.valueOf((byte) 0x40);
52  
53          assertThat(deviceGroup, nullValue());
54      }
55  
56  
57      @Test
58      @Category(FastTests.class)
59      public void disconnectReason() {
60          DisconnectReason disconnectReason = DisconnectReason.ADDRESS_UNKNOWN;
61  
62          assertThat("3 incorrectly mapped", DisconnectReason.valueOf((byte) 3), equalTo(DisconnectReason.ADDRESS_UNKNOWN));
63          assertThat(disconnectReason.getCode(), equalTo((byte) 3));
64      }

91      @Test
92      @Category(FastTests.class)
93      public void protocolClass() {
94          ProtocolClass protocolClass;
95  
96          protocolClass = ProtocolClass.CLASS_1;
97          assertThat(protocolClass.getCode(), equalTo((byte) 0x10));
98  
99          protocolClass = ProtocolClass.CLASS_2;
100         assertThat(protocolClass.getCode(), equalTo((byte) 0x20));
101 
102         protocolClass = ProtocolClass.CLASS_3;
103         assertThat(protocolClass.getCode(), equalTo((byte) 0x30));
104 
105         protocolClass = ProtocolClass.CLASS_4;
106         assertThat(protocolClass.getCode(), equalTo((byte) 0x40));
107     }

152     @Test
153     @Category(FastTests.class)
154     public void typduSize() {
155         TpduSize tpduSize = TpduSize.SIZE_128;
156 
157         assertThat(TpduSize.valueOf((byte) 0x07), equalTo(TpduSize.SIZE_128));
158         assertThat(tpduSize.getCode(), equalTo((byte) 0x07));
159         assertThat(tpduSize.getValue(), equalTo(128));
160     }

170     /**
171      * If we are requesting exactly the size of one of the iso tp
172      * pdu sizes, then exactly that box should be returned.
173      */
174     @Test
175     @Category(FastTests.class)
176     public void tpduValueForGivenExactFit() {
177         TpduSize tpduSize = TpduSize.valueForGivenSize(256);
178 
179         assertThat(tpduSize, equalTo(TpduSize.SIZE_256));
180     }
181 
182     /**
183      * In this case we have a given value that is in-between the boundaries of
184      * a pdu box, the method should return the next larger box.
185      */
186     @Test
187     @Category(FastTests.class)
188     public void tpduValueForGivenIntermediateSize() {
189         TpduSize tpduSize = TpduSize.valueForGivenSize(222);
190 
191         assertThat(tpduSize, equalTo(TpduSize.SIZE_256));
192         assertThat(tpduSize.getValue(), not(equalTo(222)));
193     }
194 
195     /**
196      * This test should cause an exception as the tpdu size has to be greater
197      * than 0 in any case.
198      */
199     @Test(expected = IllegalArgumentException.class)
200     @Category(FastTests.class)
201     public void tpduValueForGivenTooSmallSize() {
202         TpduSize.valueForGivenSize(-1);
203     }
204 
205     /**
206      * In this test the tpdu size is greater than the maximum defined by the iso tp
207      * protocol spec, so it is automatically downgraded to the maximum valid value.
208      */
209     @Test
210     @Category(FastTests.class)
211     public void tpduValueForGivenTooGreatSize() {
212         TpduSize tpduSize = TpduSize.valueForGivenSize(10000);
213         assertThat(tpduSize, equalTo(TpduSize.SIZE_8192));
214     }
