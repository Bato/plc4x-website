19  package org.apache.plc4x.java.s7.netty.model.types;
20  
21  import java.util.HashMap;
22  import java.util.Map;
23  
24  public enum ParameterError {
25      NO_ERROR((short) 0x0000),
26      INVALID_BLOCK_TYPE_NUMBER((short) 0x0110),
27      INVALID_PARAMETER((short) 0x0112),
28      PG_RESSOURCE_ERROR((short) 0x011A),
29      PLC_RESSOURCE_ERROR((short) 0x011B),
30      PROTOCOL_ERROR((short) 0x011C),
31      USER_BUFFER_TOO_SHORT((short) 0x011F),
32      REQUEST_ERROR((short) 0x0141),
33      VERSION_MISMATCH((short) 0x01C0),
34      NOT_IMPLEMENTED((short) 0x01F0),
35      L7_INVALID_CPU_STATE((short) 0x8001),
36      L7_PDU_SIZE_ERROR((short) 0x8500),
37      L7_INVALID_SZL_ID((short) 0xD401),
38      L7_INVALID_INDEX((short) 0xD402),
39      L7_DGS_CONNECTION_ALREADY_ANNOUNCED((short) 0xD403),
40      L7_MAX_USER_NB((short) 0xD404),
41      L7_DGS_FUNCTION_PARAMETER_SYNTAX_ERROR((short) 0xD405),
42      L7_NO_INFO((short) 0xD406),
43      L7_PRT_FUNCTION_PARAMETER_SYNTAX_ERROR((short) 0xD601),
44      L7_INVALID_VARIABLE_ADDRESS((short) 0xD801),
45      L7_UNKNOWN_REQUEST((short) 0xD802),
46      L7_INVALID_REQUEST_STATUS((short) 0xD803);
47  
48      private static final Map<Short, ParameterError> map;
49      static {
50          map = new HashMap<>();
51          for (ParameterError parameterError : ParameterError.values()) {
52              map.put(parameterError.code, parameterError);
53          }
54      }
55  
56      private final short code;
57  
58      ParameterError(short code) {
59          this.code = code;
60      }
61  
62      public short getCode() {
63          return code;
64      }
65  
66      public static ParameterError valueOf(short code) {
67          return map.get(code);
68      }
69  
70  }
