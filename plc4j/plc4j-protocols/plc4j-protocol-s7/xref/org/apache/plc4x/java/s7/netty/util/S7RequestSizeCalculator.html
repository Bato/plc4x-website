19  package org.apache.plc4x.java.s7.netty.util;

21  import org.apache.commons.lang3.NotImplementedException;
22  import org.apache.plc4x.java.s7.netty.model.messages.S7RequestMessage;
23  import org.apache.plc4x.java.s7.netty.model.params.S7Parameter;
24  import org.apache.plc4x.java.s7.netty.model.params.VarParameter;
25  import org.apache.plc4x.java.s7.netty.model.params.items.VarParameterItem;
26  import org.apache.plc4x.java.s7.netty.model.payloads.S7Payload;
27  import org.apache.plc4x.java.s7.netty.model.payloads.VarPayload;
28  import org.apache.plc4x.java.s7.netty.model.payloads.items.VarPayloadItem;
29  import org.apache.plc4x.java.s7.netty.model.types.VariableAddressingMode;
30  
31  import java.util.List;
32  
33  /**
34   * When sending S7 messages we have to pay attention to the size of the message
35   * in it's serialized form as we have to stay below the agreed upon maximum
36   * PDU size. This helper calculates this message size and hereby helps implementing
37   * different strategies for optimizing requests.
38   */
39  public class S7RequestSizeCalculator {
40  
41      private S7RequestSizeCalculator() {
42          // We don't want this to be instantiated.
43      }
44  
45      /**
46       * Calculate the size of a request message.
47       *
48       * @param requestMessage the request message we want to get the size for.
49       * @return the size in bytes this message will have.
50       */
51      public static short getRequestMessageSize(S7RequestMessage requestMessage) {
52          // The fixed header size.
53          short size = 10;
54          size += getRequestParametersSize(requestMessage.getParameters());
55          size += getRequestPayloadsSize(requestMessage.getPayloads());
56          return size;
57      }
58  
59      /**
60       * Calculates the size adding an item (parameter and eventually payload) would add to any existing
61       * message.
62       *
63       * @param varParameterItem parameter item.
64       * @param varPayloadItem corresponding payload (can be null).
65       * @return size in bytes this item would add to an existing request message.
66       */
67      public static short getRequestItemTotalSize(VarParameterItem varParameterItem, VarPayloadItem varPayloadItem) {
68          return (short) (getRequestReadWriteVarParameterItemSize(varParameterItem) + ((varPayloadItem != null) ?
69              getRequestWriteVarPayloadItemSize(varPayloadItem) : 0));
70      }
71  
72      ////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
73      // Internal Request size calculations.
74      ////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
75  
76      private static short getRequestParametersSize(List<S7Parameter> parameters) {
77          short l = 0;
78          if (parameters != null) {
79              for (S7Parameter parameter : parameters) {
80                  l += getRequestParameterSize(parameter);
81              }
82          }
83          return l;
84      }
85  
86      private static short getRequestParameterSize(S7Parameter parameter) {
87          switch (parameter.getType()) {
88              case READ_VAR:
89              case WRITE_VAR:
90                  return getRequestReadWriteVarParameterSize((VarParameter) parameter);
91              case SETUP_COMMUNICATION:
92                  return 8;
93              default:
94                  throw new NotImplementedException("Not implemented");
95          }
96      }
97  
98      private static short getRequestReadWriteVarParameterSize(VarParameter varParameter) {
99          // A var parameter always has a minimum size of 2 bytes (type of parameter, number of items)
100         short length = 2;
101         // Then come the items ...
102         for (VarParameterItem varParameterItem : varParameter.getItems()) {
103             length += getRequestReadWriteVarParameterItemSize(varParameterItem);
104         }
105         return length;
106     }
107 
108     private static short getRequestReadWriteVarParameterItemSize(VarParameterItem varParameterItem) {
109         VariableAddressingMode addressMode = varParameterItem.getAddressingMode();
110         // S7 Any items have a fixed size of 12 bytes.
111         if (addressMode == VariableAddressingMode.S7ANY) {
112             return 12;
113         } else {
114             throw new NotImplementedException("Not implemented");
115         }
116     }
117 
118     private static short getRequestPayloadsSize(List<S7Payload> payloads) {
119         short l = 0;
120         if (payloads != null) {
121             for (S7Payload payload : payloads) {
122                 l += getRequestPayloadSize(payload);
123             }
124         }
125         return l;
126     }
127 
128     private static short getRequestPayloadSize(S7Payload payload) {
129         switch (payload.getType()) {
130             case WRITE_VAR:
131                 return getRequestWriteVarPayloadSize((VarPayload) payload);
132             default:
133                 throw new NotImplementedException("Not implemented");
134         }
135     }
136 
137     private static short getRequestWriteVarPayloadSize(VarPayload varPayload) {
138         short length = 0;
139         // Then come the items ...
140         for (VarPayloadItem varPayloadItem : varPayload.getItems()) {
141             length += getRequestWriteVarPayloadItemSize(varPayloadItem);
142         }
143         return length;
144     }
145 
146     private static short getRequestWriteVarPayloadItemSize(VarPayloadItem varPayloadItem) {
147         // A var payload item always has a minimum size of 4 bytes (return code, transport size, size (two bytes))
148         short length = 4;
149         length += varPayloadItem.getData().length;
150         // It seems that bit payloads need a additional separating 0x00 byte.
151         if(varPayloadItem.getDataTransportSize().isSizeInBits()) {
152             length += 1;
153         }
154         return length;
155     }
