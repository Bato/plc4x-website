19  package org.apache.plc4x.java.isotp.protocol.model.types;
20  
21  import java.util.HashMap;
22  import java.util.Map;
23  
24  public enum DisconnectReason {
25      // Reasons available for all classes:
26      REASON_NOT_SPECIFIED((byte) 0x00),
27      CONGESTION_AT_TSAP((byte) 0x01),
28      SESSION_ENTITY_NOT_ATACHED_TO_TSAP((byte) 0x02),
29      ADDRESS_UNKNOWN((byte) 0x03),
30  
31      // Reasons only available for classes 1 to 4:
32      NORMAL((byte) 0x80),
33      REMOTE_TRANSPORT_ENTITY_CONGESTION((byte) 0x81),
34      CONNECTION_NEGOTIATION_FAILED((byte) 0x82),
35      DUPLICATE_SOURCE_REFERENCE((byte) 0x83),
36      MISMATCHED_REFERENCES((byte) 0x84),
37      PROTOCOL_ERROR((byte) 0x85),
38      REFERENCE_OVERFLOW((byte) 0x87),
39      CONNECTION_REQUEST_REFUSED((byte) 0x88),
40      HEADER_OR_PARAMETER_LENGTH_INVALID((byte) 0x8A);
41  
42      private static final Map<Byte, DisconnectReason> map;
43      static {
44          map = new HashMap<>();
45          for (DisconnectReason disconnectReason : DisconnectReason.values()) {
46              map.put(disconnectReason.code, disconnectReason);
47          }
48      }
49  
50      private final byte code;
51  
52      DisconnectReason(byte code) {
53          this.code = code;
54      }
55  
56      public byte getCode() {
57          return code;
58      }
59  
60      public static DisconnectReason valueOf(byte code) {
61          return map.get(code);
62      }
63  
64  }
