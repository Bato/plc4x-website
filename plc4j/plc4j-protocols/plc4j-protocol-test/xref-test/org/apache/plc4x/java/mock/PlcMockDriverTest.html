41  public class PlcMockDriverTest {

43      private final PlcMockDriver driver = new PlcMockDriver();
44  
45      @Test
46      public void fetchTwoConnections_areEqual() throws PlcConnectionException {
47          PlcConnection conn1 = driver.connect("mock:123");
48          PlcConnection conn2 = driver.connect("mock:123");
49  
50          assertEquals(conn1, conn2);
51      }
52  
53      /**
54       * Example of Mock Usage.
55       * The idea is to fetch a mock connection with a specific name and prepare a {@link MockDevice} which is set there.
56       *
57       * Some application code which uses the same Connection String will then automatically get the same connection
58       * and operate against the same {@link MockDevice} without the necessity to also mock field queries or other things.
59       *
60       * In this example the {@link #someCodeWhichShouldDoPlcManipulation(String)} function represents the Business Logic
61       * which should be tested and where only the connection string is manipulated for the test.
62       */
63      @Test
64      public void testScenarioExample() throws PlcConnectionException, ExecutionException, InterruptedException {
65          PlcMockConnection preparingConnection = ((PlcMockConnection) driver.connect("test:123"));
66          MockDevice mock = Mockito.mock(MockDevice.class);
67          when(mock.read(any())).thenReturn(Pair.of(PlcResponseCode.OK, new DefaultIntegerFieldItem(1)));
68          preparingConnection.setDevice(mock);
69  
70          // Now we can simply inject this URL into our code and automatically have our mock
71          someCodeWhichShouldDoPlcManipulation("test:123");
72  
73          // Verify that the code did indeed what we wanted it to do
74          verify(mock, times(1)).read("DB2.DBD17:INT");
75      }
76  
77      /**
78       * Example function that does some reading from a siemens plc using Siemens Syntax
79       * @param connection Connection String, e.g., from config
80       */
81      private void someCodeWhichShouldDoPlcManipulation(String connection) throws PlcConnectionException, ExecutionException, InterruptedException {
82          // Normally this would be from the driver manager
83          PlcConnection connect = driver.connect(connection);
84          PlcReadResponse response = connect.readRequestBuilder()
85              .addItem("value", "DB2.DBD17:INT")
86              .build()
87              .execute()
88              .get();
89          // Normally do something with the response
90      }
