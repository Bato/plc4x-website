44      public static byte[] encodeData(AdsDataType adsDataType, Object... values) throws PlcProtocolException {
45          if (values.length == 0) {
46              return new byte[]{};
47          }
48          Class<?> valueType = values[0].getClass();
49          Stream<byte[]> result;
50          if (valueType == Boolean.class) {
51              result = encodeBoolean(adsDataType, Arrays.stream(values).map(Boolean.class::cast));
52          } else if (valueType == Byte.class) {
53              result = encodeByte(adsDataType, Arrays.stream(values).map(Byte.class::cast));
54          } else if (valueType == Short.class) {
55              result = encodeShort(adsDataType, Arrays.stream(values).map(Short.class::cast));
56          } else if (valueType == Integer.class) {
57              result = encodeInteger(adsDataType, Arrays.stream(values).map(Integer.class::cast));
58          } else if (valueType == Long.class) {
59              result = encodeLong(adsDataType, Arrays.stream(values).map(Long.class::cast));
60          } else if (valueType == BigInteger.class) {
61              result = encodeBigInteger(adsDataType, Arrays.stream(values).map(BigInteger.class::cast));
62          } else if (valueType == LocalTime.class) {
63              result = encodeLocalTime(adsDataType, Arrays.stream(values).map(LocalTime.class::cast));
64          } else if (valueType == LocalDate.class) {
65              result = encodeLocalDate(adsDataType, Arrays.stream(values).map(LocalDate.class::cast));
66          } else if (valueType == LocalDateTime.class) {
67              result = encodeLocalDateTime(adsDataType, Arrays.stream(values).map(LocalDateTime.class::cast));
68          } else if (valueType == Float.class) {
69              result = encodeFloat(adsDataType, Arrays.stream(values).map(Float.class::cast));
70          } else if (valueType == Double.class) {
71              result = encodeDouble(adsDataType, Arrays.stream(values).map(Double.class::cast));
72          } else if (valueType == String.class) {
73              result = encodeString(adsDataType, Arrays.stream(values).map(String.class::cast));
74          } else if (valueType == byte[].class) {
75              result = encodeByteArray(adsDataType, Arrays.stream(values).map(byte[].class::cast));
76          } else if (valueType == Byte[].class) {
77              result = encodeBigByteArray(adsDataType, Arrays.stream(values).map(Byte[].class::cast));
78          } else {
79              throw new PlcUnsupportedDataTypeException(valueType);
80          }
81  
82          // TODO: maybe we can replace this by a smarter flatmap
83          try {
84              return result.collect(
85                  ByteArrayOutputStream::new,
86                  (bos, byteValue) -> {
87                      try {
88                          bos.write(byteValue);
89                      } catch (IOException e) {
90                          throw new PlcRuntimeException(e);
91                      }
92                  },
93                  (a, b) -> {
94                  }).toByteArray();
95          } catch (PlcRuntimeException e) {
96              throw new PlcProtocolException("Error encoding data", e);
97          }
98      }

100     private static Stream<byte[]> encodeString(AdsDataType adsDataType, Stream<String> stringStream) {
101         // TODO: add boundchecks and add optional extension
102         // TODO: what do we do with utf-8 values with 2 bytes? what is the charset here?
103         return stringStream
104             .map(s -> s.getBytes(Charset.defaultCharset()))
105             // TODO: this 0 termination is from s7 but might be completly wrong in ads. Guess its a terminator
106             .map(bytes -> ArrayUtils.add(bytes, (byte) 0x0));
107     }

109     private static Stream<byte[]> encodeByteArray(AdsDataType adsDataType, Stream<byte[]> byteArrayStream) {
110         // TODO: add boundchecks and add optional extension
111         return byteArrayStream
112             .peek(bytes -> {
113                 if (bytes.length > adsDataType.getTargetByteSize()) {
114                     throw new PlcRuntimeException(new PlcProtocolPayloadTooBigException("ads", adsDataType.getTargetByteSize(), bytes.length, bytes));
115                 }
116             });
117     }

119     private static Stream<byte[]> encodeBigByteArray(AdsDataType adsDataType, Stream<Byte[]> byteArrayStream) {
120         // TODO: add boundchecks and add optional extension
121         return byteArrayStream.map(ArrayUtils::toPrimitive)
122             .peek(bytes -> {
123                 if (bytes.length > adsDataType.getTargetByteSize()) {
124                     throw new PlcRuntimeException(new PlcProtocolPayloadTooBigException("ads", adsDataType.getTargetByteSize(), bytes.length, bytes));
125                 }
126             });
127     }

129     private static Stream<byte[]> encodeFloat(AdsDataType adsDataType, Stream<Float> floatStream) {
130         return floatStream
131             .peek(value -> checkBound(adsDataType, value))
132             // TODO: check how ads expects this data
133             .map(Float::floatToIntBits)
134             .map(intValue -> new byte[]{
135                 (byte) (intValue & 0x000000ff),
136                 (byte) ((intValue & 0x0000ff00) >> 8),
137                 (byte) ((intValue & 0x00ff0000) >> 16),
138                 (byte) ((intValue & 0xff000000) >> 24),
139             })
140             .map(bytes -> Arrays.copyOf(bytes, adsDataType.getTargetByteSize()));
141     }

143     private static Stream<byte[]> encodeDouble(AdsDataType adsDataType, Stream<Double> doubleStream) {
144         return doubleStream
145             .peek(value -> checkBound(adsDataType, value))
146             // TODO: check how ads expects this data
147             .map(Double::doubleToLongBits)
148             .map(longValue -> new byte[]{
149                 (byte) (longValue & 0x00000000_000000ffL),
150                 (byte) ((longValue & 0x00000000_0000ff00L) >> 8),
151                 (byte) ((longValue & 0x00000000_00ff0000L) >> 16),
152                 (byte) ((longValue & 0x00000000_ff000000L) >> 24),
153                 (byte) ((longValue & 0x000000ff_00000000L) >> 32),
154                 (byte) ((longValue & 0x0000ff00_00000000L) >> 40),
155                 (byte) ((longValue & 0x00ff0000_00000000L) >> 48),
156                 (byte) ((longValue & 0xff000000_00000000L) >> 56),
157             })
158             .map(bytes -> Arrays.copyOf(bytes, adsDataType.getTargetByteSize()));
159     }

189     private static Stream<byte[]> encodeBigInteger(AdsDataType adsDataType, Stream<BigInteger> bigIntegerStream) {
190         // TODO: add boundchecks and add optional extension
191         return bigIntegerStream
192             .map(bigIntValue -> {
193                 byte[] bytes = bigIntValue.toByteArray();
194                 if (bytes.length > 1 && bytes[0] == 0x0) {
195                     byte[] subArray = Arrays.copyOf(ArrayUtils.subarray(bytes, 1, bytes.length), adsDataType.getTargetByteSize());
196                     ArrayUtils.reverse(subArray);
197                     return subArray;
198                 } else {
199                     ArrayUtils.reverse(Arrays.copyOf(bytes, adsDataType.getTargetByteSize()));
200                     return bytes;
201                 }
202             })
203             .map(bytes -> Arrays.copyOf(bytes, adsDataType.getTargetByteSize()));
204     }

206     private static Stream<byte[]> encodeLocalTime(AdsDataType adsDataType, Stream<LocalTime> localTimeStream) {
207         return localTimeStream
208             .map(localTime -> ChronoUnit.MILLIS.between(LocalTime.of(0, 0), localTime))
209             .peek(value -> checkBound(adsDataType, value))
210             .map(Long::intValue)
211             .map(time -> new byte[]{
212                 (byte) (time & 0x00000000_000000ffL),
213                 (byte) ((time & 0x00000000_0000ff00L) >> 8),
214                 (byte) ((time & 0x00000000_00ff0000L) >> 16),
215                 (byte) ((time & 0x00000000_ff000000L) >> 24),
216             })
217             .map(bytes -> Arrays.copyOf(bytes, adsDataType.getTargetByteSize()));
218     }

220     private static Stream<byte[]> encodeLocalDate(AdsDataType adsDataType, Stream<LocalDate> localDateStream) {
221         return localDateStream
222             // TODO: fixme: which offset should we use?
223             .map(localDate -> localDate.atTime(0, 0).toInstant(ZoneOffset.UTC))
224             .map(Instant::getEpochSecond)
225             .peek(value -> checkBound(adsDataType, value))
226             .map(Long::intValue)
227             .map(time -> new byte[]{
228                 (byte) (time & 0x00000000_000000ffL),
229                 (byte) ((time & 0x00000000_0000ff00L) >> 8),
230                 (byte) ((time & 0x00000000_00ff0000L) >> 16),
231                 (byte) ((time & 0x00000000_ff000000L) >> 24),
232             })
233             .map(bytes -> Arrays.copyOf(bytes, adsDataType.getTargetByteSize()));
234     }

257     private static Stream<byte[]> encodeShort(AdsDataType adsDataType, Stream<Short> shortStream) {
258         return shortStream
259             .peek(value -> checkBound(adsDataType, value))
260             .map(shortValue -> new byte[]{
261                 (byte) (shortValue & 0x00ff),
262                 (byte) ((shortValue & 0xff00) >> 8),
263             })
264             .map(bytes -> Arrays.copyOf(bytes, adsDataType.getTargetByteSize()));
265     }

267     private static Stream<byte[]> encodeByte(AdsDataType adsDataType, Stream<Byte> byteStream) {
268         return byteStream
269             .peek(value -> checkBound(adsDataType, value))
270             .map(aByte -> new byte[]{aByte})
271             .map(bytes -> Arrays.copyOf(bytes, adsDataType.getTargetByteSize()));
272     }

274     private static Stream<byte[]> encodeBoolean(AdsDataType adsDataType, Stream<Boolean> booleanStream) {
275         return booleanStream
276             .map(booleanValue -> new byte[]{booleanValue ? (byte) 0x01 : (byte) 0x00})
277             .map(bytes -> Arrays.copyOf(bytes, adsDataType.getTargetByteSize()));
278     }
279 
280     private static void checkBound(AdsDataType adsDataType, double other) {
281         if (!adsDataType.withinBounds(other)) {
282             throw new PlcRuntimeException(other + " not within bounds of " + adsDataType);
283         }
284     }
