164     private final long hex;
165     private final long dec;
166     private final String description;
167     private final String possibleCauses;
168     private final String solution;
169 
170     AdsReturnCode(long hex, long dec, String description) {
171         this(hex, dec, description, "");
172     }
173 
174     AdsReturnCode(long hex, long dec, String description, String possibleCauses) {
175         this(hex, dec, description, possibleCauses, "");
176     }
177 
178     AdsReturnCode(long hex, long dec, String description, String possibleCauses, String solution) {
179         this.hex = hex;
180         this.dec = dec;
181         if (hex != dec) {
182             throw new IllegalArgumentException("hex " + hex + " is different from dec " + dec);
183         }
184         this.description = requireNonNull(description);
185         this.possibleCauses = possibleCauses;
186         this.solution = solution;
187     }
188 
189     // TODO: optimize with map
190     public static AdsReturnCode of(long hex) {
191         for (AdsReturnCode adsReturnCode : values()) {
192             if (adsReturnCode.hex == hex) {
193                 return adsReturnCode;
194             }
195         }
196         return UNKNOWN;
197     }
198 
199     public static AdsReturnCode of(String name) {
200         return valueOf(name);
201     }
202 
203     public static AdsReturnCode ofInt(String value) {
204         return of(Long.parseLong(value));
205     }
206 
207     public long getHex() {
208         return hex;
209     }
210 
211     public long getDec() {
212         return dec;
213     }
214 
215     public String getDescription() {
216         return description;
217     }
218 
219     public String getPossibleCauses() {
220         return possibleCauses;
221     }
222 
223     public String getSolution() {
224         return solution;
225     }
226 
227     @Override
228     public String toString() {
229         return "AdsReturnCode{" +
230             "hex=" + String.format("0x%4s", toHexString(hex)).replace(' ', '0') +
231             ", dec=" + dec +
232             ", description='" + description + '\'' +
233             ", possibleCauses='" + possibleCauses + '\'' +
234             ", solution='" + solution + '\'' +
235             '}';
236     }
