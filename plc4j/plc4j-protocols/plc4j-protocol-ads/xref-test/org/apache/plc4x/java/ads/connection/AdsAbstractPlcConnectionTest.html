137         void simulatePipelineError(FutureProducingTestRunnable futureProducingTestRunnable) {
138             ChannelFuture channelFuture = mock(ChannelFuture.class);
139             // Simulate error in the pipeline
140             when(channelFuture.addListener(any())).thenAnswer(invocation -> {
141                 Future future = mock(Future.class);
142                 when(future.isSuccess()).thenReturn(false);
143                 when(future.cause()).thenReturn(new DummyException());
144                 GenericFutureListener genericFutureListener = invocation.getArgument(0);
145                 genericFutureListener.operationComplete(future);
146                 return mock(ChannelFuture.class);
147             });
148             when(channel.writeAndFlush(any())).thenReturn(channelFuture);
149             assertThrows(DummyException.class, () -> {
150                 CompletableFuture completableFuture = futureProducingTestRunnable.run();
151                 try {
152                     completableFuture.get(3, TimeUnit.SECONDS);
153                     fail("Should have thrown a ExecutionException");
154                 } catch (ExecutionException e) {
155                     if (e.getCause() instanceof DummyException) {
156                         throw (DummyException) e.getCause();
157                     }
158                     throw e;
159                 }
160             });
161         }

271         @Test
272         void getFromFuture() throws Exception {
273             runInThread(() -> {
274                 CompletableFuture completableFuture = mock(CompletableFuture.class, RETURNS_DEEP_STUBS);
275                 Object fromFuture = SUT.getFromFuture(completableFuture, 1);
276                 assertNotNull(fromFuture);
277             });
278             runInThread(() -> {
279                 CompletableFuture completableFuture = mock(CompletableFuture.class, RETURNS_DEEP_STUBS);
280                 when(completableFuture.get(anyLong(), any())).thenThrow(InterruptedException.class);
281                 assertThrows(PlcRuntimeException.class, () -> SUT.getFromFuture(completableFuture, 1));
282             });
283             runInThread(() -> {
284                 CompletableFuture completableFuture = mock(CompletableFuture.class, RETURNS_DEEP_STUBS);
285                 when(completableFuture.get(anyLong(), any())).thenThrow(ExecutionException.class);
286                 assertThrows(PlcRuntimeException.class, () -> SUT.getFromFuture(completableFuture, 1));
287             });
288             runInThread(() -> {
289                 CompletableFuture completableFuture = mock(CompletableFuture.class, RETURNS_DEEP_STUBS);
290                 when(completableFuture.get(anyLong(), any())).thenThrow(TimeoutException.class);
291                 assertThrows(PlcRuntimeException.class, () -> SUT.getFromFuture(completableFuture, 1));
292             });
293             assertFalse("The current Thread should not be interrupted", Thread.currentThread().isInterrupted());
294         }

296         /**
297          * Runs tests steps in a dedicated {@link Thread} so a possible {@link InterruptedException} doesn't lead to a
298          * interrupt flag being set on the main Thread ({@link Thread#isInterrupted}).
299          *
300          * @param testRunnable a special {@link Runnable} which adds a {@code throws Exception} to the {@code run} signature.
301          * @throws InterruptedException when this {@link Thread} gets interrupted.
302          */
303         void runInThread(TestRunnable testRunnable) throws InterruptedException {
304             Thread thread = new Thread(() -> {
305                 try {
306                     testRunnable.run();
307                 } catch (Exception e) {
308                     throw new RuntimeException(e);
309                 }
310             });
311             Queue<Throwable> uncaughtExceptions = new ConcurrentLinkedQueue<>();
312             thread.setUncaughtExceptionHandler((t, e) -> uncaughtExceptions.add(e));
313             thread.start();
314             thread.join();
315             if (!uncaughtExceptions.isEmpty()) {
316                 uncaughtExceptions.forEach(throwable -> LOGGER.error("Assertion Error: Unexpected Exception", throwable));
317                 throw new AssertionError("Test failures. Check log");
318             }
319         }
320     }

322     /**
323      * Variant of {@link Runnable} which adds a {@code throws Exception} to the {@code run} signature.
324      */
325     private interface TestRunnable {
326         /**
327          * @throws Exception when the test throws a exception.
328          * @see Runnable#run()
329          */
330         void run() throws Exception;
331     }
