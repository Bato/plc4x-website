19  package org.apache.plc4x.java.ads;
20  
21  import org.apache.plc4x.java.PlcDriverManager;
22  import org.apache.plc4x.java.api.PlcConnection;
23  import org.apache.plc4x.java.api.messages.*;
24  import org.apache.plc4x.java.api.model.PlcConsumerRegistration;
25  
26  import java.time.Duration;
27  import java.util.Collection;
28  import java.util.List;
29  import java.util.concurrent.CompletableFuture;
30  import java.util.concurrent.TimeUnit;
31  import java.util.stream.Collectors;
32  
33  public class ManualPlc4XAdsTest {
34  
35      public static void main(String... args) throws Exception {
36          String connectionUrl;
37          if (args.length > 0 && "serial".equalsIgnoreCase(args[0])) {
38              System.out.println("Using serial");
39              connectionUrl = "ads:serial:///dev/ttys003/10.10.64.40.1.1:851/10.10.56.23.1.1:30000";
40          } else {
41              System.out.println("Using tcp");
42              connectionUrl = "ads:tcp://10.10.64.40/10.10.64.40.1.1:851/10.10.56.23.1.1:30000";
43          }
44          // TODO: temporary workaround
45          Thread.currentThread().setUncaughtExceptionHandler((t, e) -> {
46              System.err.println(t + " - " + e.getMessage());
47              e.printStackTrace(System.err);
48              System.exit(1);
49          });
50          try (PlcConnection plcConnection = new PlcDriverManager().getConnection(connectionUrl)) {
51              System.out.println("PlcConnection " + plcConnection);
52  
53              PlcReadRequest.Builder readRequestBuilder = plcConnection.readRequestBuilder();
54              PlcReadRequest readRequest = readRequestBuilder.addItem("station", "Allgemein_S2.Station:BYTE").build();
55              CompletableFuture<? extends PlcReadResponse> response = readRequest.execute();
56              PlcReadResponse readResponse = response.get();
57              System.out.println("Response " + readResponse);
58              Collection<Integer> stations = readResponse.getAllIntegers("station");
59              stations.forEach(System.out::println);
60  
61              // 2. We build a subscription
62              PlcSubscriptionRequest.Builder subscriptionRequestBuilder = plcConnection.subscriptionRequestBuilder();
63              PlcSubscriptionRequest subscriptionRequest = subscriptionRequestBuilder
64                  .addChangeOfStateField("stationChange", "Allgemein_S2.Station:BYTE")
65                  .addCyclicField("stationChange2", "Allgemein_S2.Station:BYTE", Duration.ofSeconds(3))
66                  .build();
67              PlcSubscriptionResponse plcSubscriptionResponse = subscriptionRequest.execute().get();
68  
69              List<PlcConsumerRegistration> plcConsumerRegistrations = plcSubscriptionResponse.getSubscriptionHandles().stream()
70                  .map(plcSubscriptionHandle -> plcSubscriptionHandle.register(plcSubscriptionEvent -> {
71                      int stationChange = plcSubscriptionEvent.getNumberOfValues("stationChange");
72                      int stationChange2 = plcSubscriptionEvent.getNumberOfValues("stationChange2");
73                      System.out.println(String.format("%s: [%d]- StationsNummer: {%d}", plcSubscriptionEvent.getTimestamp(), stationChange + stationChange2, plcSubscriptionEvent.getInteger("stationChange2")));
74                  }))
75                  .collect(Collectors.toList());
76  
77              // Now we wait a bit
78              TimeUnit.SECONDS.sleep(10);
79  
80              // we unregister the listener
81              plcConsumerRegistrations.forEach(PlcConsumerRegistration::unregister);
82  
83              // we unsubscribe at the plc
84              PlcUnsubscriptionRequest.Builder unsubscriptionRequestBuilder = plcConnection.unsubscriptionRequestBuilder();
85              PlcUnsubscriptionRequest unsubscriptionRequest = unsubscriptionRequestBuilder.addHandles(plcSubscriptionResponse.getSubscriptionHandles()).build();
86              CompletableFuture<PlcUnsubscriptionResponse> unsubscriptionResponse = unsubscriptionRequest.execute();
87  
88              unsubscriptionResponse
89                  .get(5, TimeUnit.SECONDS);
90              System.out.println(unsubscriptionResponse);
91          }
92          System.exit(0);
93      }
94  }
