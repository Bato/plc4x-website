19  package org.apache.plc4x.java.spi;
20  
21  import org.apache.plc4x.java.api.PlcConnection;
22  import org.apache.plc4x.java.api.authentication.PlcAuthentication;
23  import org.apache.plc4x.java.api.exceptions.PlcConnectionException;
24  
25  /**
26   * General interface defining the minimal methods required for adding a new type of driver to the PLC4J system.
27   *
28   * <b>Note that each driver has to add a service file called org.apache.plc4x.java.spi.PlcDriver to
29   * src/main/resources/META-INF which contains the fully qualified classname in order to get loaded
30   * by the PlcDriverManager instances.</b>
31   */
32  public interface PlcDriver {
33  
34      /**
35       * @return code of the implemented protocol. This is usually a lot shorter than the String returned by @seeÂ #getProtocolName().
36       */
37      String getProtocolCode();
38  
39      /**
40       * @return name of the implemented protocol.
41       */
42      String getProtocolName();
43  
44      /**
45       * Connects to a PLC using the given plc connection string.
46       * @param url plc connection string.
47       * @return PlcConnection object.
48       * @throws PlcConnectionException an exception if the connection attempt failed.
49       */
50      PlcConnection connect(String url) throws PlcConnectionException;
51  
52      /**
53       * Connects to a PLC using the given plc connection string using given authentication credentials.
54       * @param url plc connection string.
55       * @param authentication authentication credentials.
56       * @return PlcConnection object.
57       * @throws PlcConnectionException an exception if the connection attempt failed.
58       */
59      PlcConnection connect(String url, PlcAuthentication authentication) throws PlcConnectionException;
60  
61  }
