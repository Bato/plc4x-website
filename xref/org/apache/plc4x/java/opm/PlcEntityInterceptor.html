54  /**
55   * Interceptor for dynamic functionality of @{@link PlcEntity}.
56   * Basically, its {@link #intercept(Object, Method, Callable, String, PlcDriverManager, AliasRegistry, Map)} method is called for each
57   * invocation of a method on a connected @{@link PlcEntity} and does then the dynamic part.
58   *
59   * For those not too familiar with the JVM's dispatch on can roughly imagine the intercept method being a "regular"
60   * method on the "proxied" entity and all parameters of the intercept method could then be access to local fields.
61   *
62   * @author julian
63   */
64  public class PlcEntityInterceptor {
65  
66      private static final Logger LOGGER = LoggerFactory.getLogger(PlcEntityInterceptor.class);
67  
68      private static final Configuration CONF = new SystemConfiguration();
69      private static final long READ_TIMEOUT = CONF.getLong("org.apache.plc4x.java.opm.entity_manager.read_timeout", 1_000);

71      private PlcEntityInterceptor() {
72          throw new UnsupportedOperationException("This class is not to be instantiated");
73      }

75      /**
76       * Basic Intersector for all methods on the proxy object.
77       * It checks if the invoked method is a getter and if so, only retrieves the requested field, forwarding to
78       * the {@link #fetchValueForGetter(Object, Method, PlcDriverManager, String, AliasRegistry, Map)} method.
79       * <p>
80       * If the field is no getter, then all fields are refreshed by calling {@link #refetchAllFields(Object, PlcDriverManager, String, AliasRegistry, Map)}
81       * and then, the method is invoked.
82       *
83       * @param proxy    Object to intercept
84       * @param method   Method that was intercepted
85       * @param callable Callable to call the method after fetching the values
86       * @param address  Address of the plc (injected from private field)
87       * @param driverManager DriverManager instance to use (injected from private field)
88       * @return possible result of the original methods invocation
89       * @throws OPMException Problems with plc / proxying
90       */
91      @SuppressWarnings("unused")
92      @RuntimeType
93      public static Object intercept(@This Object proxy, @Origin Method method, @SuperCall Callable<?> callable,
94                                     @FieldValue(PlcEntityManager.PLC_ADDRESS_FIELD_NAME) String address,
95                                     @FieldValue(PlcEntityManager.DRIVER_MANAGER_FIELD_NAME) PlcDriverManager driverManager,
96                                     @FieldValue(PlcEntityManager.ALIAS_REGISTRY) AliasRegistry registry,
97                                     @FieldValue(PlcEntityManager.LAST_FETCHED) Map<String, Instant> lastFetched) throws OPMException {
98          LOGGER.trace("Invoked method {} on connected PlcEntity {}", method.getName(), method.getDeclaringClass().getName());
99  
100         // If "detached" (i.e. _driverManager is null) simply forward the call
101         if (driverManager == null) {
102             LOGGER.trace("Entity not connected, simply fowarding call");
103             try {
104                 return callable.call();
105             } catch (Exception e) {
106                 throw new OPMException("Exception during forwarding call", e);
107             }
108         }
109 
110         if (method.getName().startsWith("get")) {
111             if (method.getParameterCount() > 0) {
112                 throw new OPMException("Only getter with no arguments are supported");
113             }
114             // Fetch single value
115             LOGGER.trace("Invoked method {} is getter, trying to find annotated field and return requested value",
116                 method.getName());
117 
118             return fetchValueForGetter(proxy, method, driverManager, address, registry, lastFetched);
119         }
120 
121         if (method.getName().startsWith("is") && (method.getReturnType() == boolean.class || method.getReturnType() == Boolean.class)) {
122             if (method.getParameterCount() > 0) {
123                 throw new OPMException("Only getter with no arguments are supported");
124             }
125             // Fetch single value
126             LOGGER.trace("Invoked method {} is boolean flag method, trying to find annotated field and return requested value",
127                 method.getName());
128             return fetchValueForIsGetter(proxy, method, driverManager, address, registry, lastFetched);
129         }
130 
131         // Fetch all values, than invoke method
132         try {
133             LOGGER.trace("Invoked method is no getter, refetch all fields and invoke method {} then", method.getName());
134             refetchAllFields(proxy, driverManager, address, registry, lastFetched);
135             return callable.call();
136         } catch (Exception e) {
137             throw new OPMException("Unable to forward invocation " + method.getName() + " on connected PlcEntity", e);
138         }
139     }

141     /**
142      * Renews all values of all Fields that are annotated with {@link PlcEntity}.
143      *
144      * @param proxy Object to refresh the fields on.
145      * @param driverManager Driver Manager to use
146      * @param registry AliasRegistry to use
147      * @param lastFetched
148      * @throws OPMException on various errors.
149      */
150     @SuppressWarnings("squid:S1141") // Nested try blocks readability is okay, move to other method makes it imho worse
151     static void refetchAllFields(Object proxy, PlcDriverManager driverManager, String address, AliasRegistry registry, Map<String, Instant> lastFetched) throws OPMException {
152         // Don't log o here as this would cause a second request against a plc so don't touch it, or if you log be aware of that
153         Class<?> entityClass = proxy.getClass().getSuperclass();
154         LOGGER.trace("Refetching all fields on proxy object of class {}", entityClass);
155         PlcEntity plcEntity = entityClass.getAnnotation(PlcEntity.class);
156         if (plcEntity == null) {
157             throw new OPMException("Non PlcEntity supplied");
158         }
159 
160         // Check if all fields are valid
161         for (Field field : entityClass.getDeclaredFields()) {
162             if (field.isAnnotationPresent(PlcField.class)) {
163                 OpmUtils.getOrResolveAddress(registry, field.getAnnotation(PlcField.class).value());
164             }
165         }
166         try (PlcConnection connection = driverManager.getConnection(address)) {
167             // Catch the exception, if no reader present (see below)
168             // Build the query
169             PlcReadRequest.Builder requestBuilder = connection.readRequestBuilder();
170 
171             Arrays.stream(entityClass.getDeclaredFields())
172                 .filter(field -> field.isAnnotationPresent(PlcField.class))
173                 .filter(field -> needsToBeFetched(lastFetched, field))
174                 .forEach(field ->
175                     requestBuilder.addItem(
176                         getFqn(field),
177                         OpmUtils.getOrResolveAddress(registry, field.getAnnotation(PlcField.class).value())
178                     )
179                 );
180 
181             PlcReadRequest request = requestBuilder.build();
182 
183             LOGGER.trace("Request for refetch of {} was build and is {}", entityClass, request);
184 
185             PlcReadResponse response = getPlcReadResponse(request);
186 
187             // Fill all requested fields
188             for (String fieldName : response.getFieldNames()) {
189                 // Fill into Cache
190                 lastFetched.put(fieldName, Instant.now());
191 
192                 LOGGER.trace("Value for field {}  is {}", fieldName, response.getObject(fieldName));
193                 String clazzFieldName = StringUtils.substringAfterLast(fieldName, ".");
194                 try {
195                     setField(entityClass, proxy, response, clazzFieldName, fieldName);
196                 } catch (NoSuchFieldException | IllegalAccessException e) {
197                     throw new PlcRuntimeException(e);
198                 }
199             }
200         } catch (PlcConnectionException e) {
201             throw new OPMException("Problem during processing", e);
202         } catch (Exception e) {
203             throw new OPMException("Unexpected error during processing", e);
204         }
205     }
206 
207     private static String getFqn(Field field) {
208         return field.getDeclaringClass().getName() + "." + field.getName();
209     }
210 
211     /**
212      * Checks if a field needs to be refetched, i.e., the cached values are too old.
213      */
214     private static boolean needsToBeFetched(Map<String, Instant> lastFetched, Field field) {
215         Validate.notNull(field);
216         long cacheDurationMillis = field.getAnnotation(PlcField.class).cacheDurationMillis();
217         String fqn = getFqn(field);
218         if (lastFetched.containsKey(fqn)) {
219             Instant last = lastFetched.get(fqn);
220             return Instant.now().minus(cacheDurationMillis, ChronoUnit.MILLIS).isAfter(last);
221         }
222         return true;
223     }

229     private static Object fetchValueForGetter(Object proxy, Method m, PlcDriverManager driverManager, String address, AliasRegistry registry, Map<String, Instant> lastFetched) throws OPMException {
230         return fetchValueForGetter(proxy, m, 3, driverManager, address, registry, lastFetched);

233     private static Object fetchValueForGetter(Object proxy, Method m, int prefixLength, PlcDriverManager driverManager,
234                                               String address, AliasRegistry registry, Map<String, Instant> lastFetched) throws OPMException {
235         String s = m.getName().substring(prefixLength);
236         // First char to lower
237         String variable = s.substring(0, 1).toLowerCase().concat(s.substring(1));
238         LOGGER.trace("Looking for field with name {} after invokation of getter {}", variable, m.getName());
239         PlcField annotation;
240         Field field;
241         try {
242             field = m.getDeclaringClass().getDeclaredField(variable);
243             annotation = field.getDeclaredAnnotation(PlcField.class);
244         } catch (NoSuchFieldException e) {
245             throw new OPMException("Unable to identify field with name '" + variable + "' for call to '" + m.getName() + "'", e);
246         }
247 
248         // Use Fully qualified Name as field index
249         String fqn = getFqn(field);
250 
251         // Check if cache is still active
252         if (!needsToBeFetched(lastFetched, field)) {
253             // Return the current value
254             try {
255                 field.setAccessible(true);
256                 return field.get(proxy);
257             } catch (IllegalAccessException e) {
258                 throw new OPMException("Unable to restore cached (previous) value for field '" + field.getName() + "'", e);
259             }
260         }
261         try (PlcConnection connection = driverManager.getConnection(address)) {
262             // Catch the exception, if no reader present (see below)
263 
264             PlcReadRequest request = connection.readRequestBuilder()
265                 .addItem(fqn, OpmUtils.getOrResolveAddress(registry, annotation.value()))
266                 .build();
267 
268             PlcReadResponse response = getPlcReadResponse(request);
269 
270             // Fill into Cache
271             lastFetched.put(field.getName(), Instant.now());
272 
273             return getTyped(m.getReturnType(), response, fqn);
274         } catch (ClassCastException e) {
275             throw new OPMException("Unable to return response as suitable type", e);
276         } catch (Exception e) {
277             throw new OPMException("Problem during processing", e);
278         }
279     }

281     /**
282      * Tries to set a response Item to a field in the given object.
283      * This is one by looking for a field in the class and a response item
284      * which is equal to the given fieldName parameter.
285      *
286      * @param o               Object to set the value on
287      * @param response        Response to fetch the response from
288      * @param targetFieldName Name of the field in the object
289      * @param sourceFieldName Name of the field in the response
290      * @throws NoSuchFieldException If a field is not present in entity
291      * @throws IllegalAccessException If a field in the entity cannot be accessed
292      */
293     static void setField(Class<?> clazz, Object o, PlcReadResponse response, String targetFieldName, String sourceFieldName) throws NoSuchFieldException, IllegalAccessException {
294         LOGGER.debug("setField on clazz: {}, Object: {}, response: {}, targetFieldName: {}, sourceFieldName:{} ", clazz, o, response, targetFieldName, sourceFieldName);
295         Field field = clazz.getDeclaredField(targetFieldName);
296         field.setAccessible(true);
297         try {
298             field.set(o, getTyped(field.getType(), response, sourceFieldName));
299         } catch (ClassCastException e) {
300             throw new PlcRuntimeException(String.format("Unable to assign return value %s to field %s with type %s",
301                 response.getObject(sourceFieldName), targetFieldName, field.getType()), e);
302         }
303     }

305     @SuppressWarnings({"squid:S3776", "squid:MethodCyclomaticComplexity"}) // Cognitive Complexity not too high, as highly structured
306     static Object getTyped(Class<?> clazz, PlcReadResponse response, String sourceFieldName) {
307         LOGGER.debug("getTyped clazz: {}, response: {}, fieldName: {}", clazz, response, sourceFieldName);
308         if (response.getResponseCode(sourceFieldName) != PlcResponseCode.OK) {
309             throw new PlcRuntimeException(String.format("Unable to read specified field '%s', response code was '%s'",
310                 sourceFieldName, response.getResponseCode(sourceFieldName)));
311         }
312         if (clazz.isPrimitive()) {
313             if (clazz == boolean.class) {
314                 return response.getBoolean(sourceFieldName);
315             } else if (clazz == byte.class) {
316                 return response.getByte(sourceFieldName);
317             } else if (clazz == short.class) {
318                 return response.getShort(sourceFieldName);
319             } else if (clazz == int.class) {
320                 return response.getInteger(sourceFieldName);
321             } else if (clazz == long.class) {
322                 return response.getLong(sourceFieldName);
323             }
324         }
325 
326         if (clazz == Boolean.class) {
327             return response.getBoolean(sourceFieldName);
328         } else if (clazz == Byte.class) {
329             return response.getByte(sourceFieldName);
330         } else if (clazz == Short.class) {
331             return response.getShort(sourceFieldName);
332         } else if (clazz == Integer.class) {
333             return response.getInteger(sourceFieldName);
334         } else if (clazz == Long.class) {
335             return response.getLong(sourceFieldName);
336         } else if (clazz == BigInteger.class) {
337             return response.getBigInteger(sourceFieldName);
338         } else if (clazz == Float.class) {
339             return response.getFloat(sourceFieldName);
340         } else if (clazz == Double.class) {
341             return response.getDouble(sourceFieldName);
342         } else if (clazz == BigDecimal.class) {
343             return response.getBigDecimal(sourceFieldName);
344         } else if (clazz == String.class) {
345             return response.getString(sourceFieldName);
346         } else if (clazz == LocalTime.class) {
347             return response.getTime(sourceFieldName);
348         } else if (clazz == LocalDate.class) {
349             return response.getDate(sourceFieldName);
350         } else if (clazz == LocalDateTime.class) {
351             return response.getDateTime(sourceFieldName);
352         } else if (clazz == byte[].class) {
353             return ArrayUtils.toPrimitive(response.getByteArray(sourceFieldName));
354         } else if (clazz == Byte[].class) {
355             return response.getByteArray(sourceFieldName);
356         }
357 
358         // Fallback
359         Object responseObject = response.getObject(sourceFieldName);
360         if (clazz.isAssignableFrom(responseObject.getClass())) {
361             return responseObject;
362         }
363 
364         // If nothing matched, throw
365         throw new ClassCastException("Unable to return response item " + responseObject + "(" + responseObject.getClass() + ") as instance of " + clazz);
366     }

368     /**
369      * Fetch the request and do appropriate error handling
370      *
371      * @param request the request to get the exception from
372      * @return the response from the exception.
373      * @throws OPMException on {@link InterruptedException} or {@link ExecutionException} or {@link TimeoutException}
374      */
375     static PlcReadResponse getPlcReadResponse(PlcReadRequest request) throws OPMException {
376         try {
377             return request.execute().get(READ_TIMEOUT, TimeUnit.MILLISECONDS);
378         } catch (InterruptedException e) {
379             Thread.currentThread().interrupt();
380             throw new OPMException("Exception during execution", e);
381         } catch (ExecutionException e) {
382             throw new OPMException("Exception during execution", e);
383         } catch (TimeoutException e) {
384             throw new OPMException("Timeout during fetching values", e);
385         }
386     }
