63      @Override
64      protected void encode(ChannelHandlerContext ctx, PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> msg, List<Object> out) throws Exception {
65          LOGGER.trace("(<--OUT): {}, {}, {}", ctx, msg, out);
66          // Reset transactionId on overflow
67          transactionId.compareAndSet(Short.MAX_VALUE + 1, 0);
68          PlcRequest request = msg.getRequest();
69          if (request instanceof PlcReadRequest) {
70              encodeReadRequest(msg, out);
71          } else if (request instanceof PlcWriteRequest) {
72              encodeWriteRequest(msg, out);
73          }
74      }

76      private void encodeWriteRequest(PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> msg, List<Object> out) throws PlcException {
77          InternalPlcWriteRequest request = (InternalPlcWriteRequest) msg.getRequest();
78  
79          // TODO: support multiple requests
80          if (request.getFieldNames().size() != 1) {
81              throw new PlcNotImplementedException("Only single message supported for now");
82          }
83          // TODO: check if we can map like this. Implication is that we can only work with int, short, byte and boolean
84          // TODO: for higher data types float, double etc we might need to split the bytes into chunks
85          String fieldName = request.getFieldNames().iterator().next();
86          int quantity = request.getNumberOfValues(fieldName);
87          ModbusField field = (ModbusField) request.getField(fieldName);
88          if (quantity != field.getQuantity()) {
89              LOGGER.warn("Supplied number of values [{}] don't match t the addressed quantity of [{}]", field.getQuantity(), quantity);
90          }
91  
92          short unitId = 0;
93  
94          /*
95           * It seems that in Modbus, there are only two types of resources, that can be accessed:
96           * - Register: 2 byte value
97           * - Coil: 1 bit value
98           *
99           * Registers:
100          * When writing a bit, byte or char (a one byte or less data types) into a register the value is filled up to
101          * fit the 2 bytes.
102          * When writing a data type that has more than 2 bytes, subsequent registers are written to automatically.
103          *
104          * Coils:
105          * When transferring data from/to a coil, 8 coil values can be transferred in a single byte.
106          * Naturally a coil is a boolean data type, however, similar to the registers bigger data types can be
107          * transported, by setting multiple subsequent coils: 32bit int -> 32 coils
108          *
109          * In all cases where we are accessing more than the natural size of the datatype, we have to keep this in mind
110          * when addressing them. So if reading 2 32bit integers, this is split up into four registers. So for the second
111          * int we have to increment the address accordingly.
112          */
113         ModbusPdu modbusRequest;
114         if (field instanceof RegisterModbusField) {
115             RegisterModbusField registerModbusField = (RegisterModbusField) field;
116             if (quantity > 1) {
117                 byte[] bytesToWrite = produceRegisterValue(Arrays.asList(request.getFieldItem(fieldName).getValues()));
118                 // A register is a 16 bit (2 byte) value ... so every value needs 2 byte.
119                 int requiredLength = 2 * quantity;
120                 if (bytesToWrite.length != requiredLength) {
121                     throw new PlcProtocolException("Invalid register values created. Should be at least quantity * 2 = N bytes. Was " + bytesToWrite.length + ", expected " + requiredLength);
122                 }
123                 modbusRequest = new WriteMultipleRegistersRequest(registerModbusField.getAddress(), quantity, bytesToWrite);
124             } else {
125                 byte[] register = produceRegisterValue(Arrays.asList(request.getFieldItem(fieldName).getValues()));
126                 if ((register == null) || (register.length != 2)) {
127                     throw new PlcProtocolException("Invalid register values created. Should be 2 bytes. Was " +
128                         ((register != null) ? register.length : 0));
129                 }
130                 // Reconvert the two bytes back to an int.
131                 int intToWrite = register[0] << 8 | register[1] & 0xff;
132                 modbusRequest = new WriteSingleRegisterRequest(registerModbusField.getAddress(), intToWrite);
133             }
134         } else if (field instanceof CoilModbusField) {
135             CoilModbusField coilModbusField = (CoilModbusField) field;
136             if (quantity > 1) {
137                 byte[] bytesToWrite = produceCoilValues(Arrays.asList(request.getFieldItem(fieldName).getValues()));
138                 // As each coil value represents a bit, the number of bytes needed
139                 // equals "ceil(quantity/8)" (a 3 bit shift is a division by 8 ... the +1 is the "ceil")
140                 int requiredLength = (quantity >> 3) + 1;
141                 if (bytesToWrite.length != requiredLength) {
142                     throw new PlcProtocolException(
143                         "Invalid coil values created. Should be big enough to transport N bits. Was " +
144                             bytesToWrite.length + ", expected " + requiredLength);
145                 }
146                 modbusRequest = new WriteMultipleCoilsRequest(coilModbusField.getAddress(), quantity, bytesToWrite);
147             } else {
148                 boolean booleanToWrite = produceCoilValue(Arrays.asList(request.getFieldItem(fieldName).getValues()));
149                 modbusRequest = new WriteSingleCoilRequest(coilModbusField.getAddress(), booleanToWrite);
150             }
151         } else if (field instanceof MaskWriteRegisterModbusField) {
152             MaskWriteRegisterModbusField maskWriteRegisterModbusField = (MaskWriteRegisterModbusField) field;
153             if (quantity > 1) {
154                 throw new PlcProtocolException("Mask write request can only write one value");
155             } else {
156                 // TODO: this should be better part of the payload not the addressing.
157                 int andMask = maskWriteRegisterModbusField.getAndMask();
158                 int orMask = maskWriteRegisterModbusField.getOrMask();
159                 modbusRequest = new MaskWriteRegisterRequest(maskWriteRegisterModbusField.getAddress(), andMask, orMask);
160             }
161         } else {
162             throw new PlcProtocolException("Unsupported field type " + field.getClass() + " for a write request.");
163         }
164         short transactionId = (short) this.transactionId.getAndIncrement();
165         requestsMap.put(transactionId, msg);
166         out.add(new ModbusTcpPayload(transactionId, unitId, modbusRequest));
167     }

169     private void encodeReadRequest(PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> msg, List<Object> out) throws PlcException {
170         PlcReadRequest request = (PlcReadRequest) msg.getRequest();
171         // TODO: support multiple requests
172         if (request.getFieldNames().size() != 1) {
173             throw new PlcNotImplementedException("Only single message supported for now");
174         }
175         // TODO: check if we can map like this. Implication is that we can only work with int, short, byte and boolean
176         // TODO: for higher data types float, double etc we might need to split the bytes into chunks
177         String fieldName = request.getFieldNames().iterator().next();
178 
179         ModbusField field = (ModbusField) request.getField(fieldName);
180         int quantity = field.getQuantity();
181         // TODO: the unit the should be used for multiple Requests
182         short unitId = 0;
183 
184         ModbusPdu modbusRequest;
185         if (field instanceof CoilModbusField) {
186             CoilModbusField coilModbusField = (CoilModbusField) field;
187             modbusRequest = new ReadCoilsRequest(coilModbusField.getAddress(), quantity);
188         } else if (field instanceof RegisterModbusField) {
189             RegisterModbusField registerModbusField = (RegisterModbusField) field;
190             modbusRequest = new ReadHoldingRegistersRequest(registerModbusField.getAddress(), quantity);
191         } else if (field instanceof ReadDiscreteInputsModbusField) {
192             ReadDiscreteInputsModbusField readDiscreteInputsModbusField = (ReadDiscreteInputsModbusField) field;
193             modbusRequest = new ReadDiscreteInputsRequest(readDiscreteInputsModbusField.getAddress(), quantity);
194         } else if (field instanceof ReadHoldingRegistersModbusField) {
195             ReadHoldingRegistersModbusField readHoldingRegistersModbusField = (ReadHoldingRegistersModbusField) field;
196             modbusRequest = new ReadHoldingRegistersRequest(readHoldingRegistersModbusField.getAddress(), quantity);
197         } else if (field instanceof ReadInputRegistersModbusField) {
198             ReadInputRegistersModbusField readInputRegistersModbusField = (ReadInputRegistersModbusField) field;
199             modbusRequest = new ReadInputRegistersRequest(readInputRegistersModbusField.getAddress(), quantity);
200         } else {
201             throw new PlcProtocolException("Unsupported field type " + field.getClass() + " for a read request.");
202         }
203         short transactionId = (short) this.transactionId.getAndIncrement();
204         requestsMap.put(transactionId, msg);
205         out.add(new ModbusTcpPayload(transactionId, unitId, modbusRequest));
206     }

213         // TODO: implement me
214         short transactionId = msg.getTransactionId();
215         PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> plcRequestContainer = requestsMap.get(transactionId);
216         if (plcRequestContainer == null) {
217             throw new PlcProtocolException("Unrelated payload received. [transactionId: " + msg.getTransactionId() + ", unitId: " + msg.getUnitId() + ", modbusPdu: " + msg.getModbusPdu() + "]");
218         }
219

323     private boolean produceCoilValue(List<?> values) throws PlcProtocolException {
324         if (values.size() != 1) {
325             throw new PlcProtocolException("Only one value allowed");
326         }
327         byte multiCoil = produceCoilValues(values)[0];
328         return multiCoil != 0;
329     }
330 
331     private byte[] produceCoilValues(List<?> values) throws PlcProtocolException {
332         List<Byte> coils = new LinkedList<>();
333         byte actualCoil = 0;
334         int i = 7;
335         for (Object value : values) {
336             final boolean coilSet;
337             if (value.getClass() == Boolean.class) {
338                 coilSet = (Boolean) value;
339             } else if (value.getClass() == Byte.class) {
340                 if ((byte) value > 1) {
341                     throw new PlcProtocolException("Value to high to fit into Byte: " + value);
342                 }
343                 coilSet = (byte) value == 1;
344             } else if (value.getClass() == byte[].class) {
345                 byte[] bytes = (byte[]) value;
346                 if (bytes.length != 1) {
347                     throw new PlcProtocolException("Exactly one byte is allowed: " + bytes.length);
348                 }
349                 byte byteValue = bytes[0];
350                 if (byteValue > 1) {
351                     throw new PlcProtocolException("Value to high to fit into Byte: " + value);
352                 }
353                 coilSet = byteValue == 1;
354             } else if (value.getClass() == Byte[].class) {
355                 Byte[] bytes = (Byte[]) value;
356                 if (bytes.length != 1) {
357                     throw new PlcProtocolException("Exactly one byte is allowed: " + bytes.length);
358                 }
359                 byte byteValue = bytes[0];
360                 if (byteValue > 1) {
361                     throw new PlcProtocolException("Value to high to fit into Byte: " + value);
362                 }
363                 coilSet = byteValue == 1;
364             } else if (value.getClass() == Short.class) {
365                 if ((short) value > 1) {
366                     throw new PlcProtocolException("Value to high to fit into Byte: " + value);
367                 }
368                 coilSet = (Short) value == 1;
369             } else if (value.getClass() == Integer.class) {
370                 if ((int) value > 1) {
371                     throw new PlcProtocolException("Value to high to fit into Byte: " + value);
372                 }
373                 coilSet = (int) value == 1;
374             } else if (value.getClass() == BigInteger.class) {
375                 coilSet = value.equals(BigInteger.ONE);
376             } else if (value.getClass() == Float.class) {
377                 coilSet = value.equals(1.0f);
378             } else if (value.getClass() == Double.class) {
379                 coilSet = value.equals(1.0d);
380             } else {
381                 throw new PlcUnsupportedDataTypeException(value.getClass());
382             }
383             byte coilToSet = coilSet ? (byte) 1 : (byte) 0;
384             actualCoil = (byte) (actualCoil & 0xff | coilToSet << i);
385             i--;
386             if (i < 0) {
387                 coils.add(actualCoil);
388                 actualCoil = 0;
389                 i = 8;
390             }
391         }
392         if (coils.isEmpty()) {
393             // We only have one coil
394             return new byte[]{actualCoil};
395         }
396         return ArrayUtils.toPrimitive(coils.toArray(new Byte[0]));
397     }

399     private byte[] produceRegisterValue(List<?> values) throws PlcProtocolException {
400         ByteBuf buffer = Unpooled.buffer();
401         long upperRegisterValue = 0xFFFFL;
402         for (Object value : values) {
403             if (value.getClass() == Boolean.class) {
404                 buffer.writeByte(0);
405                 buffer.writeByte((boolean) value ? 1 : 0);
406             } else if (value.getClass() == Byte.class) {
407                 buffer.writeByte(0);
408                 buffer.writeByte((byte) value);
409             } else if (value.getClass() == byte[].class) {
410                 byte[] bytes = (byte[]) value;
411                 if (bytes.length != 2) {
412                     throw new PlcProtocolException("Exactly two bytes are allowed: " + bytes.length);
413                 }
414                 buffer.writeBytes(bytes);
415             } else if (value.getClass() == Byte[].class) {
416                 Byte[] bytes = (Byte[]) value;
417                 if (bytes.length != 2) {
418                     throw new PlcProtocolException("Exactly two bytes are allowed: " + bytes.length);
419                 }
420                 buffer.writeBytes(ArrayUtils.toPrimitive(bytes));
421             } else if (value.getClass() == Short.class) {
422                 if ((short) value < 0) {
423                     throw new PlcProtocolException("Only positive values are supported for Short: " + value);
424                 }
425                 buffer.writeShort((short) value);
426             } else if (value.getClass() == Integer.class) {
427                 if ((int) value > upperRegisterValue) {
428                     throw new PlcProtocolException("Value to high to fit into register for Integer: " + value);
429                 }
430                 if ((int) value < 0) {
431                     throw new PlcProtocolException("Only positive values are supported for Integer: " + value);
432                 }
433                 buffer.writeShort((int) value);
434             } else if (value.getClass() == BigInteger.class) {
435                 if (((BigInteger) value).compareTo(BigInteger.ZERO) < 0) {
436                     throw new PlcProtocolException("Only positive values are supported for BigInteger: " + value);
437                 }
438                 if (((BigInteger) value).compareTo(BigInteger.valueOf(0XFFFF_FFFFL)) > 0) {
439                     throw new PlcProtocolException("Value to high to fit into register for BigInteger: " + value);
440                 }
441                 // TODO: for now we can't support big values as we only write one register at once
442                 if (((BigInteger) value).compareTo(BigInteger.valueOf(upperRegisterValue)) > 0) {
443                     throw new PlcProtocolException("Value to high to fit into register for BigInteger: " + value);
444                 }
445                 // TODO: Register has 2 bytes so we trim to 2 instead of 4 like the second if above
446                 int maxBytes = 2;
447                 byte[] bigIntegerBytes = ((BigInteger) value).toByteArray();
448                 byte[] bytes = new byte[maxBytes];
449                 int lengthToCopy = Math.min(bigIntegerBytes.length, maxBytes);
450                 int srcPosition = Math.max(bigIntegerBytes.length - maxBytes, 0);
451                 int destPosition = maxBytes - lengthToCopy;
452                 System.arraycopy(bigIntegerBytes, srcPosition, bytes, destPosition, lengthToCopy);
453 
454                 // TODO: check if this is a good representation.
455                 // TODO: can a big integer span multiple registers?
456                 buffer.writeBytes(bytes);
457             } else if (value.getClass() == Float.class) {
458                 if (((float) value) < 0) {
459                     throw new PlcProtocolException("Only positive values are supported for Float: " + value);
460                 }
461                 if (((float) value) > upperRegisterValue) {
462                     throw new PlcProtocolException("Value to high to fit into register for Float: " + value);
463                 }
464                 buffer.writeShort(Math.round((float) value));
465             } else if (value.getClass() == Double.class) {
466                 if (((double) value) < 0) {
467                     throw new PlcProtocolException("Only positive values are supported for Double: " + value);
468                 }
469                 if (((double) value) > upperRegisterValue) {
470                     throw new PlcProtocolException("Value to high to fit into register for Double: " + value);
471                 }
472                 buffer.writeShort((int) Math.round((double) value));
473             } else {
474                 throw new PlcUnsupportedDataTypeException(value.getClass());
475             }
476         }
477         byte[] result = new byte[buffer.writerIndex()];
478         buffer.readBytes(result);
479         return result;
480     }

485     private DefaultBooleanFieldItem produceCoilValueList(ByteBuf byteBuf, int expectedQuantity) {
486         if (byteBuf.readableBytes() < expectedQuantity / 8) {
487             LOGGER.warn("Expected to read {} coils but only max of {} can be supplied", expectedQuantity, byteBuf.readableBytes() * 8);
488         }
489         byte[] bytes = new byte[byteBuf.readableBytes()];
490         if (bytes.length < 1) {
491             return new DefaultBooleanFieldItem();
492         }
493         byteBuf.readBytes(bytes);
494         List<Boolean> data = new LinkedList<>();
495         int bitIndex = 0;
496         int coilIndex = 0;
497         while (coilIndex < bytes.length && data.size() < expectedQuantity) {
498             if (bitIndex > 7) {
499                 // Every 8 Coils we need to increase the access
500                 coilIndex++;
501                 bitIndex = 0;
502                 if (coilIndex >= bytes.length) {
503                     break;
504                 }
505             }
506             boolean coilSet = (bytes[coilIndex] & 0xff & (1L << bitIndex)) != 0;
507             data.add(coilSet);
508             bitIndex++;
509         }
510         return new DefaultBooleanFieldItem(data.toArray(new Boolean[0]));
511     }

513     private DefaultModbusByteArrayFieldItem produceRegisterValueList(ByteBuf byteBuf, int expectedQuantity) throws PlcProtocolException {
514         int readableBytes = byteBuf.readableBytes();
515         if (readableBytes % 2 != 0) {
516             throw new PlcProtocolException("Readables bytes should even: " + readableBytes);
517         }
518         List<Byte[]> data = new LinkedList<>();
519         while (byteBuf.readableBytes() > 0) {
520             byte[] register = new byte[2];
521             byteBuf.readBytes(register);
522             data.add(ArrayUtils.toObject(register));
523         }
524         return new DefaultModbusByteArrayFieldItem(data.toArray(new Byte[0][0]));
525     }
