19  package org.apache.plc4x.java.ads.api.generic;
20  
21  import io.netty.buffer.ByteBuf;
22  import org.apache.plc4x.java.ads.api.commands.AdsCommandType;
23  import org.apache.plc4x.java.ads.api.generic.types.*;
24  import org.apache.plc4x.java.ads.api.util.ByteReadable;
25  
26  import static java.util.Objects.requireNonNull;
27  
28  public abstract class AmsPacket implements ByteReadable {
29  
30      protected final AmsHeader amsHeader;
31  
32      protected AmsPacket(AmsHeader amsHeader) {
33          this.amsHeader = amsHeader;
34      }
35  
36      protected AmsPacket(AmsNetId targetAmsNetId, AmsPort targetAmsPort, AmsNetId sourceAmsNetId, AmsPort sourceAmsPort, State stateId, Invoke invokeId) {
37          if (!getClass().isAnnotationPresent(AdsCommandType.class)) {
38              throw new IllegalArgumentException(AdsCommandType.class + " need to be present.");
39          }
40          this.amsHeader = AmsHeader.of(
41              requireNonNull(targetAmsNetId),
42              requireNonNull(targetAmsPort),
43              requireNonNull(sourceAmsNetId),
44              requireNonNull(sourceAmsPort),
45              requireNonNull(getClass().getAnnotation(AdsCommandType.class).value()),
46              requireNonNull(stateId),
47              () -> getAdsData().getCalculatedLength(),
48              requireNonNull(AmsError.NONE),
49              requireNonNull(invokeId));
50      }
51  
52      public AmsHeader getAmsHeader() {
53          return amsHeader;
54      }
55  
56      public abstract AdsData getAdsData();
57  
58      protected AdsData buildADSData(ByteReadable... byteReadables) {
59          return () -> buildByteBuff(byteReadables);
60      }
61  
62      @Override
63      public ByteBuf getByteBuf() {
64          return buildByteBuff(amsHeader, getAdsData());
65      }
66  
67      @Override
68      public boolean equals(Object o) {
69          if (this == o) {
70              return true;
71          }
72          if (!(o instanceof AmsPacket)) {
73              return false;
74          }
75  
76          AmsPacket amsPacket = (AmsPacket) o;
77  
78          return amsHeader.equals(amsPacket.amsHeader);
79      }
80  
81      @Override
82      public int hashCode() {
83          return amsHeader.hashCode();
84      }
85  
86      @Override
87      public String toString() {
88          return getClass().getSimpleName() + "{" +
89              "amsHeader=" + amsHeader +
90              '}';
91      }
92  }
