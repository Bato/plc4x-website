60  /**
61   * Communication Layer between the Application level ({@link Plc4XS7Protocol}) and the lower level (tcp) that sends and receives {@link S7Message}s.
62   * This layer also handles the control over the "wire", i.e., the queues of incoming and outgoing messages.
63   * Furthermore, here {@link S7Message}s are marshalled and unmarshalled to {@link ByteBuf}s to be send over wire.
64   *
65   * Before messages are send to the wire an optional {@link S7MessageProcessor} can be applied.
66   *
67   * @see S7MessageProcessor
68   */
69  public class S7Protocol extends ChannelDuplexHandler {

110     @Override
111     public void channelRegistered(ChannelHandlerContext ctx) {
112         this.queue = new PendingWriteQueue(ctx);
113         try {
114             Field prevField = FieldUtils.getField(ctx.getClass(), "prev", true);
115             if(prevField != null) {
116                 ChannelHandlerContext prevContext = (ChannelHandlerContext) prevField.get(ctx);
117                 prevChannelHandler = prevContext.handler();
118             }
119         } catch(Exception e) {
120             logger.error("Error accessing field 'prev'", e);
121         }
122     }
123 
124     @Override
125     public void channelUnregistered(ChannelHandlerContext ctx) throws Exception {
126         this.queue.removeAndWriteAll();
127         super.channelUnregistered(ctx);
128     }
129 
130     @Override
131     public void channelInactive(ChannelHandlerContext ctx) throws Exception {
132         // Send everything so we get a proper failure for those pending writes
133         this.queue.removeAndWriteAll();
134         super.channelInactive(ctx);
135     }

137     /**
138      * If the S7 protocol layer is used over Iso TP, then after receiving a {@link IsoTPConnectedEvent} the
139      * corresponding S7 setup communication message has to be sent in order to negotiate the S7 protocol layer.
140      *
141      * @param ctx the current protocol layers context
142      * @param evt the event
143      * @throws Exception throws an exception if something goes wrong internally
144      */
145     @Override
146     public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
147         // If we are using S7 inside of IsoTP, then we need to intercept IsoTPs connected events.
148         if ((prevChannelHandler instanceof IsoTPProtocol) && (evt instanceof IsoTPConnectedEvent)) {
149             // Setup Communication
150             SetupCommunicationRequestMessage setupCommunicationRequest =
151                 new SetupCommunicationRequestMessage((short) 0, maxAmqCaller, maxAmqCallee, pduSize, null);
152 
153             ctx.channel().writeAndFlush(setupCommunicationRequest);
154         }
155 
156         else {
157             super.userEventTriggered(ctx, evt);
158         }
159     }

165     @Override
166     public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise) {
167         try {
168             if(msg instanceof S7Message) {
169                 S7Message in = (S7Message) msg;
170 
171                 // Give message processors to process the incoming message.
172                 Collection<? extends S7Message> messages;
173                 if ((messageProcessor != null) && (in instanceof S7RequestMessage)) {
174                     messages = messageProcessor.processRequest((S7RequestMessage) in, pduSize);
175                 } else {
176                     messages = Collections.singleton(in);
177                 }
178 
179                 // Create a promise that has to be called multiple times.
180                 PromiseCombiner promiseCombiner = new PromiseCombiner();
181                 for (S7Message message : messages) {
182                     ByteBuf buf = Unpooled.buffer();
183                     writeS7Message(promise.channel(), promiseCombiner, message, buf);
184                 }
185                 promiseCombiner.finish(promise);
186 
187                 // Start sending the queue content.
188                 trySendingMessages(ctx);
189             }
190             // Especially during the phase of connection establishment, we might be sending
191             // messages of a lower level protocol, so if it's not S7, we forward it to the next
192             // in the pipeline and hope it can handle it. If no layer can handle it Netty will
193             // exceptionally complete the future.
194             else {
195                 ctx.write(msg, promise);
196             }
197         } catch (Exception e) {
198             promise.setFailure(e);
199         }
200     }

202     private void writeS7Message(Channel channel, PromiseCombiner promiseCombiner,
203                                 S7Message message, ByteBuf buf) throws PlcProtocolException {
204         encodeHeader(message, buf);
205         encodeParameters(message, buf);
206         encodePayloads(message, buf);
207 
208         // Check if the message doesn't exceed the negotiated maximum size.
209         if (buf.writerIndex() > pduSize) {
210             throw new PlcProtocolPayloadTooBigException("s7", pduSize, buf.writerIndex(), message);
211         } else {
212             ChannelPromise subPromise = new DefaultChannelPromise(channel);
213             queue.add(new DataTpdu(true, (byte) 0x01, Collections.emptyList(), buf, message), subPromise);
214             promiseCombiner.add((Future) subPromise);
215             logger.debug("S7 Message with id {} queued", message.getTpduReference());
216         }
217     }

219     private void encodePayloads(S7Message in, ByteBuf buf) throws PlcProtocolException {
220         if(in.getPayloads() != null) {
221             for (S7Payload payload : in.getPayloads()) {
222                 switch (payload.getType()) {
223                     case WRITE_VAR:
224                         encodeWriteVarPayload((VarPayload) payload, buf);
225                         break;
226                     case CPU_SERVICES:
227                         encodeCpuServicesPayload((CpuServicesPayload) payload, buf);
228                         break;
229                     default:
230                         throw new PlcProtocolException("Writing payloads of type " +
231                             payload.getType().name() + " not implemented.");
232                 }
233             }
234         }
235     }

237     private void encodeWriteVarPayload(VarPayload varPayload, ByteBuf buf) {
238         for (VarPayloadItem payloadItem : varPayload.getItems()) {
239             buf.writeByte(payloadItem.getReturnCode().getCode());
240             buf.writeByte(payloadItem.getDataTransportSize().getCode());
241             // TODO: Check if this is correct?!?! Might be problems with sizeInBits = true/false
242             buf.writeShort(payloadItem.getData().length);
243             buf.writeBytes(payloadItem.getData());
244             // TODO: It looks as if BIT type reads require a 0x00 fill byte at the end ...
245         }
246     }

293     private void encodeParameters(S7Message in, ByteBuf buf) throws PlcProtocolException {
294         for (S7Parameter s7Parameter : in.getParameters()) {
295             buf.writeByte(s7Parameter.getType().getCode());
296             switch (s7Parameter.getType()) {
297                 case READ_VAR:
298                 case WRITE_VAR:
299                     encodeParameterReadWriteVar(buf, (VarParameter) s7Parameter);
300                     break;
301                 case SETUP_COMMUNICATION:
302                     encodeParameterSetupCommunication(buf, (SetupCommunicationParameter) s7Parameter);
303                     break;
304                 case CPU_SERVICES:
305                     encodeCpuServicesParameter(buf, (CpuServicesParameter) s7Parameter);
306                     break;
307                 default:
308                     throw new PlcProtocolException("Writing parameters of type " +
309                         s7Parameter.getType().name() + " not implemented.");
310             }
311         }
312     }

314     private void encodeHeader(S7Message in, ByteBuf buf) {
315         buf.writeByte(S7_PROTOCOL_MAGIC_NUMBER);
316         buf.writeByte(in.getMessageType().getCode());
317         // Reserved (is always constant 0x0000)
318         buf.writeShort((short) 0x0000);
319         // PDU Reference (Request Id, generated by the initiating node)
320         buf.writeShort(in.getTpduReference());
321         // S7 message parameters length
322         buf.writeShort(S7SizeHelper.getParametersLength(in.getParameters()));
323         // Data field length
324         buf.writeShort(S7SizeHelper.getPayloadsLength(in.getPayloads()));
325         // Not sure why this is implemented, we should never be sending out responses.
326         /*if (in instanceof S7ResponseMessage) {
327             S7ResponseMessage s7ResponseMessage = (S7ResponseMessage) in;
328             buf.writeByte(s7ResponseMessage.getErrorClass());
329             buf.writeByte(s7ResponseMessage.getErrorCode());
330         }*/
331     }

333     private void encodeParameterSetupCommunication(ByteBuf buf, SetupCommunicationParameter s7Parameter) {
334         // Reserved (is always constant 0x00)
335         buf.writeByte((byte) 0x00);
336         buf.writeShort(s7Parameter.getMaxAmqCaller());
337         buf.writeShort(s7Parameter.getMaxAmqCallee());
338         buf.writeShort(s7Parameter.getPduLength());

341     private void encodeParameterReadWriteVar(ByteBuf buf, VarParameter s7Parameter) throws PlcProtocolException {
342         List<VarParameterItem> items = s7Parameter.getItems();
343         // PlcReadRequestItem count (Read one variable at a time)
344         buf.writeByte((byte) items.size());
345         for (VarParameterItem item : items) {
346             VariableAddressingMode addressMode = item.getAddressingMode();
347             if (addressMode == VariableAddressingMode.S7ANY) {
348                 encodeS7AnyParameterItem(buf, (S7AnyVarParameterItem) item);
349             } else {
350                 throw new PlcProtocolException("Writing VarParameterItems with addressing mode " +
351                     addressMode.name() + " not implemented");
352             }
353         }
354     }

356     private void encodeCpuServicesParameter(ByteBuf buf, CpuServicesParameter parameter) {
357         // Output the header for a CPU Services parameter.
358         buf.writeByte(0x01);
359         buf.writeByte(0x12);
360         // Length of the parameter.
361         buf.writeByte((parameter instanceof CpuServicesRequestParameter) ? 0x04 : 0x08);
362         // Is this a request or a response?
363         buf.writeByte((parameter instanceof CpuServicesRequestParameter) ? 0x11 : 0x12);
364         // This is a mixture of request/response and function group .
365         byte nextByte = (byte) (((parameter instanceof CpuServicesRequestParameter) ?
366             (byte) 0x40 : (byte) 0x80) | parameter.getFunctionGroup().getCode());
367         buf.writeByte(nextByte);
368         buf.writeByte(parameter.getSubFunctionGroup().getCode());
369         buf.writeByte(parameter.getSequenceNumber());
370 
371         // A response parameter has some more fields.
372         // Not sure why this is implemented, we should never be sending out responses.

374             CpuServicesResponseParameter responseParameter = (CpuServicesResponseParameter) parameter;

381     private void encodeS7AnyParameterItem(ByteBuf buf, S7AnyVarParameterItem s7AnyRequestItem) {
382         buf.writeByte(s7AnyRequestItem.getSpecificationType().getCode());
383         // Length of this item (excluding spec type and length)
384         buf.writeByte((byte) 0x0a);
385         buf.writeByte(s7AnyRequestItem.getAddressingMode().getCode());
386         buf.writeByte(s7AnyRequestItem.getDataType().getTypeCode());
387         buf.writeShort(s7AnyRequestItem.getNumElements());
388         buf.writeShort(s7AnyRequestItem.getDataBlockNumber());
389         buf.writeByte(s7AnyRequestItem.getMemoryArea().getCode());
390         // A S7 address is 3 bytes long. Unfortunately the byte-offset is NOT located in
391         // byte 1 and byte 2 and the bit offset in byte 3. Siemens used the last 3 bits of
392         // byte 3 for the bit-offset and the remaining 5 bits of byte 3 to contain the lowest
393         // 5 bits of the byte-offset. The highest 5 bits of byte 1 are probably left unused
394         // for future extensions.
395         buf.writeShort((short) (s7AnyRequestItem.getByteOffset() >> 5));
396         buf.writeByte((byte) ((
397                 (s7AnyRequestItem.getByteOffset() & 0x1F) << 3)
398                 | (s7AnyRequestItem.getBitOffset() & 0x07)));
399     }

411     protected void decode(ChannelHandlerContext ctx, IsoTPMessage in, List<Object> out) {
412         if (logger.isTraceEnabled()) {
413             logger.trace("Got Data: {}", ByteBufUtil.hexDump(in.getUserData()));
414         }
415         ByteBuf userData = in.getUserData();
416         if (userData.readableBytes() == 0) {
417             return;
418         }
419 
420         if (userData.readByte() != S7_PROTOCOL_MAGIC_NUMBER) {
421             logger.warn("Expecting S7 protocol magic number.");
422             if (logger.isDebugEnabled()) {
423                 logger.debug("Got Data: {}", ByteBufUtil.hexDump(userData));
424             }
425             return;
426         }
427 
428         MessageType messageType = MessageType.valueOf(userData.readByte());
429         boolean isResponse = messageType == MessageType.ACK_DATA;
430         userData.readShort();  // Reserved (is always constant 0x0000)
431         short tpduReference = userData.readShort();
432         short headerParametersLength = userData.readShort();
433         short userDataLength = userData.readShort();
434         byte errorClass = 0;
435         byte errorCode = 0;
436         if (isResponse) {
437             errorClass = userData.readByte();
438             errorCode = userData.readByte();
439         }
440 
441         List<S7Parameter> s7Parameters = new LinkedList<>();
442         int i = 0;
443 
444         while (i < headerParametersLength) {
445             S7Parameter parameter = decodeParameter(userData, isResponse);
446             s7Parameters.add(parameter);
447             if (parameter instanceof SetupCommunicationParameter) {
448                 handleSetupCommunications(ctx, (SetupCommunicationParameter) parameter);
449             }
450             i += S7SizeHelper.getParameterLength(parameter);
451         }
452 
453         List<S7Payload> s7Payloads = decodePayloads(userData, isResponse, userDataLength, s7Parameters);
454 
455         logger.debug("S7 Message with id {} received", tpduReference);
456 
457         if (isResponse) {
458             S7ResponseMessage responseMessage = new S7ResponseMessage(
459                 messageType, tpduReference, s7Parameters, s7Payloads, errorClass, errorCode);
460 
461             // Remove the current response from the list of unconfirmed messages.
462             DataTpdu requestTpdu = sentButUnacknowledgedTpdus.remove(tpduReference);
463 
464             // Get the corresponding request message.
465             S7RequestMessage requestMessage = (requestTpdu != null) ? (S7RequestMessage) requestTpdu.getParent() : null;
466 
467             if(requestMessage != null) {
468                 // Set this individual request to "acknowledged".
469                 requestMessage.setAcknowledged(true);
470 
471                 // Give the request and response to a message processor to process the incoming message.
472                 if(messageProcessor != null) {
473                     try {
474                         responseMessage = messageProcessor.processResponse(requestMessage, responseMessage);
475                     } catch(Exception e) {
476                         logger.error("Error processing message", e);
477                         ctx.fireExceptionCaught(e);
478                         return;
479                     }
480                 }
481 
482                 if(responseMessage != null) {
483                     out.add(responseMessage);
484 
485                     // If this is a USER_DATA packet the probability is high that this is
486                     // a response to the identification request, we have to handle that.
487                     if(responseMessage.getMessageType() == MessageType.USER_DATA) {
488                         for (S7Payload payload : responseMessage.getPayloads()) {
489                             if(payload instanceof CpuServicesPayload) {
490                                 handleIdentifyRemote(ctx, (CpuServicesPayload) payload);
491                             }
492                         }
493                     }
494                 }
495 
496                 // Eventually send the next message (if there is one).
497                 trySendingMessages(ctx);
498             }
499 
500         } else {
501             // CpuService responses are encoded as requests.
502             for (S7Parameter s7Parameter : s7Parameters) {
503                 // Only if we have a response parameter, the payload is a response payload.
504                 if(s7Parameter instanceof CpuServicesResponseParameter) {
505                     for (S7Payload s7Payload : s7Payloads) {
506                         if(s7Payload instanceof CpuServicesPayload) {
507                             CpuServicesPayload cpuServicesPayload = (CpuServicesPayload) s7Payload;
508                             handleIdentifyRemote(ctx, cpuServicesPayload);
509                         }
510                     }
511                 }
512             }
513             out.add(new S7RequestMessage(messageType, tpduReference, s7Parameters, s7Payloads, null));
514         }
515     }

517     private void handleSetupCommunications(ChannelHandlerContext ctx, SetupCommunicationParameter setupCommunicationParameter) {
518         maxAmqCaller = setupCommunicationParameter.getMaxAmqCaller();
519         maxAmqCallee = setupCommunicationParameter.getMaxAmqCallee();
520         pduSize = setupCommunicationParameter.getPduLength();
521 
522         logger.info("S7Connection established pdu-size {}, max-amq-caller {}, " +
523                 "max-amq-callee {}", pduSize, maxAmqCaller, maxAmqCallee);
524 
525         // Prepare a message to request the remote to identify itself.
526         S7RequestMessage identifyRemoteMessage = new S7RequestMessage(MessageType.USER_DATA, (short) 2,
527             Collections.singletonList(new CpuServicesRequestParameter(
528                 CpuServicesParameterFunctionGroup.CPU_FUNCTIONS,
529                 CpuServicesParameterSubFunctionGroup.READ_SSL, (byte) 0)),
530             Collections.singletonList(new CpuServicesPayload(DataTransportErrorCode.OK, SslId.MODULE_IDENTIFICATION,
531                 (short) 0x0000)), null);
532         ctx.channel().writeAndFlush(identifyRemoteMessage);
533     }

535     private void handleIdentifyRemote(ChannelHandlerContext ctx, CpuServicesPayload cpuServicesPayload) {
536         controllerType = S7ControllerType.S7_ANY;
537         for (SslDataRecord sslDataRecord : cpuServicesPayload.getSslDataRecords()) {
538             if(sslDataRecord instanceof SslModuleIdentificationDataRecord) {
539                 SslModuleIdentificationDataRecord sslModuleIdentificationDataRecord =
540                     (SslModuleIdentificationDataRecord) sslDataRecord;
541                 if(sslModuleIdentificationDataRecord.getIndex() == (short) 0x0001) {
542                     controllerType = lookupControllerType(sslModuleIdentificationDataRecord.getArticleNumber());
543                 }
544             }
545         }
546         if(logger.isInfoEnabled()) {
547             logger.info(String.format("Successfully connected to S7: %s", controllerType.name()));
548             logger.info(String.format("- max amq caller: %s", maxAmqCaller));
549             logger.info(String.format("- max amq callee: %s", maxAmqCallee));
550             logger.info(String.format("- pdu size: %s", pduSize));
551         }
552 
553         // Send an event that connection setup is complete.
554         ctx.channel().pipeline().fireUserEventTriggered(new S7ConnectedEvent());
555     }

557     private List<S7Payload> decodePayloads(ByteBuf userData, boolean isResponse, short userDataLength, List<S7Parameter> s7Parameters) {
558         List<S7Payload> s7Payloads = new LinkedList<>();
559         for (S7Parameter s7Parameter : s7Parameters) {
560             if(s7Parameter instanceof VarParameter) {
561                 VarParameter readWriteVarParameter = (VarParameter) s7Parameter;
562                 VarPayload varPayload = decodeVarPayload(userData, isResponse, userDataLength, readWriteVarParameter);
563                 s7Payloads.add(varPayload);
564             } else if(s7Parameter instanceof CpuServicesParameter) {
565                 CpuServicesPayload cpuServicesPayload = decodeCpuServicesPayload(userData);
566                 s7Payloads.add(cpuServicesPayload);
567             }
568         }
569         return s7Payloads;
570     }

572     private VarPayload decodeVarPayload(ByteBuf userData, boolean isResponse, short userDataLength,
573                                         VarParameter readWriteVarParameter) {
574         List<VarPayloadItem> payloadItems = new LinkedList<>();
575 
576         // Just keep on reading payloads until the provided length is read.
577         int i = 0;
578         while (i < userDataLength) {
579             DataTransportErrorCode dataTransportErrorCode = DataTransportErrorCode.valueOf(userData.readByte());
580             // This is a response to a WRITE_VAR request (It only contains the return code for every sent item.
581             if ((readWriteVarParameter.getType() == ParameterType.WRITE_VAR) && isResponse) {
582                 // Initialize a rudimentary payload (This is updated in the Plc4XS7Protocol class
583                 VarPayloadItem payload = new VarPayloadItem(dataTransportErrorCode, null, null);
584                 payloadItems.add(payload);
585                 i += 1;
586             }
587             // This is a response to a READ_VAR request.
588             else if ((readWriteVarParameter.getType() == ParameterType.READ_VAR) && isResponse) {
589                 DataTransportSize dataTransportSize = DataTransportSize.valueOf(userData.readByte());
590                 short length = dataTransportSize.isSizeInBits() ?
591                     (short) Math.ceil(userData.readShort() / 8.0) : userData.readShort();
592                 byte[] data = new byte[length];
593                 userData.readBytes(data);
594                 // Initialize a rudimentary payload (This is updated in the Plc4XS7Protocol class
595                 VarPayloadItem payload = new VarPayloadItem(dataTransportErrorCode, dataTransportSize, data);
596                 payloadItems.add(payload);
597                 i += S7SizeHelper.getPayloadLength(payload);
598                 // It seems that datatype BIT reads an additional byte, if it's not the last.
599 
600                 if(dataTransportSize.isHasBlankByte() && (userData.readableBytes() > 0)) {
601                     userData.readByte();
602                     i++;
603                 }
604             }
605         }
606 
607         return new VarPayload(readWriteVarParameter.getType(), payloadItems);
608     }

610     private CpuServicesPayload decodeCpuServicesPayload(ByteBuf userData) {
611         DataTransportErrorCode returnCode = DataTransportErrorCode.valueOf(userData.readByte());
612         DataTransportSize dataTransportSize = DataTransportSize.valueOf(userData.readByte());
613         if(dataTransportSize != DataTransportSize.OCTET_STRING) {
614             // TODO: Output an error.
615         }
616         short length = userData.readShort();
617         SslId sslId = SslId.valueOf(userData.readShort());
618         short sslIndex = userData.readShort();
619         // If the length is 4 there is no `partial list length in bytes` and `partial list count` parameters.
620         if(length == 4) {
621             return new CpuServicesPayload(returnCode, sslId, sslIndex);
622         }
623         // If the length is not 4, then it has to be at least 8.
624         else if(length >= 8) {
625             // TODO: We should probably ensure we don't read more than this.
626             // Skip the partial list length in words.
627             userData.skipBytes(2);
628             short partialListCount = userData.readShort();
629             List<SslDataRecord> sslDataRecords = new LinkedList<>();
630             for(int i = 0; i < partialListCount; i++) {
631                 short index = userData.readShort();
632                 byte[] articleNumberBytes = new byte[20];
633                 userData.readBytes(articleNumberBytes);
634                 String articleNumber = new String(articleNumberBytes, StandardCharsets.UTF_8).trim();
635                 short bgType = userData.readShort();
636                 short moduleOrOsVersion = userData.readShort();
637                 short pgDescriptionFileVersion = userData.readShort();
638                 sslDataRecords.add(new SslModuleIdentificationDataRecord(
639                     index, articleNumber, bgType, moduleOrOsVersion, pgDescriptionFileVersion));
640             }
641             return new CpuServicesPayload(returnCode, sslId, sslIndex, sslDataRecords);
642         }
643         // In all other cases, it's probably an error.
644         else {
645             // TODO: Output an error.
646         }
647         return null;
648     }

650     private S7Parameter decodeParameter(ByteBuf in, boolean isResponse) {
651         ParameterType parameterType = ParameterType.valueOf(in.readByte());
652         if (parameterType == null) {
653             logger.error("Could not find parameter type");
654             return null;
655         }
656         switch (parameterType) {
657             case CPU_SERVICES:
658                 return decodeCpuServicesParameter(in);
659             case READ_VAR:
660             case WRITE_VAR:
661                 List<VarParameterItem> varParamameter;
662                 byte numItems = in.readByte();
663                 if (!isResponse) {
664                     varParamameter = decodeReadWriteVarParameter(in, numItems);
665                 } else {
666                     varParamameter = Collections.emptyList();
667                 }
668                 return new VarParameter(parameterType, varParamameter);
669             case SETUP_COMMUNICATION:
670                 // Reserved (is always constant 0x00)
671                 in.readByte();
672                 short callingMaxAmq = in.readShort();
673                 short calledMaxAmq = in.readShort();
674                 short pduLength = in.readShort();
675                 return new SetupCommunicationParameter(callingMaxAmq, calledMaxAmq, pduLength);
676             default:
677                 if (logger.isErrorEnabled()) {
678                     logger.error("Unimplemented parameter type: {}", parameterType.name());
679                 }
680         }
681         return null;
682     }

684     private CpuServicesParameter decodeCpuServicesParameter(ByteBuf in) {
685         if(in.readShort() != 0x0112) {
686             if (logger.isErrorEnabled()) {
687                 logger.error("Expecting 0x0112 for CPU_SERVICES parameter");
688             }
689             return null;
690         }
691         byte parameterLength = in.readByte();
692         if((parameterLength != 4) && (parameterLength != 8)) {
693             if (logger.isErrorEnabled()) {
694                 logger.error("Parameter length should be 4 or 8, but was {}", parameterLength);
695             }
696             return null;
697         }
698         // Skipping this as it sort of contains redundant information.
699         in.readByte();
700         byte typeAndFunctionGroup = in.readByte();
701         // If bit 7 is set, it's a request (if bit 8 is set it's a response).
702         boolean requestParameter = (typeAndFunctionGroup & 0x64) != 0;
703         // The last 4 bits contain the function group value.
704         typeAndFunctionGroup = (byte) (typeAndFunctionGroup & 0xF);
705         CpuServicesParameterFunctionGroup functionGroup =
706             CpuServicesParameterFunctionGroup.valueOf(typeAndFunctionGroup);
707         CpuServicesParameterSubFunctionGroup subFunctionGroup =
708             CpuServicesParameterSubFunctionGroup.valueOf(in.readByte());
709         byte sequenceNumber = in.readByte();
710         if(!requestParameter) {
711             return new CpuServicesRequestParameter(functionGroup, subFunctionGroup, sequenceNumber);
712         } else {
713             byte dataUnitReferenceNumber = in.readByte();
714             boolean lastDataUnit = in.readByte() == 0x00;
715             ParameterError error = ParameterError.valueOf(in.readShort());
716             return new CpuServicesResponseParameter(functionGroup, subFunctionGroup, sequenceNumber,
717                 dataUnitReferenceNumber, lastDataUnit, error);
718         }
719     }

721     private List<VarParameterItem> decodeReadWriteVarParameter(ByteBuf in, byte numItems) {
722         List<VarParameterItem> items = new LinkedList<>();
723         for (int i = 0; i < numItems; i++) {
724             SpecificationType specificationType = SpecificationType.valueOf(in.readByte());
725             // Length of the rest of this item.
726             byte itemLength = in.readByte();
727             if (itemLength != 0x0a) {
728                 logger.warn("Expecting a length of 10 here.");
729                 return items;
730             }
731             VariableAddressingMode variableAddressingMode = VariableAddressingMode.valueOf(in.readByte());
732             if (variableAddressingMode == VariableAddressingMode.S7ANY) {
733                 TransportSize dataType = TransportSize.valueOf(in.readByte());
734                 short length = in.readShort();
735                 short dbNumber = in.readShort();
736                 MemoryArea memoryArea = MemoryArea.valueOf(in.readByte());
737                 short byteAddress = (short) (in.readShort() << 5);
738                 byte tmp = in.readByte();
739                 // Only the least 3 bits are the bit address, the
740                 byte bitAddress = (byte) (tmp & 0x07);
741                 // Bits 4-8 belong to the byte address
742                 byteAddress = (short) (byteAddress | (tmp >> 3));
743                 S7AnyVarParameterItem item = new S7AnyVarParameterItem(
744                         specificationType, memoryArea, dataType,
745                         length, dbNumber, byteAddress, bitAddress);
746                 items.add(item);
747             } else {
748                 logger.error("Error parsing item type");
749                 return items;
750             }
751         }
752 
753         return items;
754     }

761     private synchronized void trySendingMessages(ChannelHandlerContext ctx) {
762         while(sentButUnacknowledgedTpdus.size() < maxAmqCaller) {
763             // Get the TPDU that is up next in the queue.
764             DataTpdu curTpdu = (DataTpdu) queue.current();
765 
766             if (curTpdu != null) {
767                 // Send the TPDU.
768                 try {
769                     ChannelFuture channelFuture = queue.removeAndWrite();
770                     ctx.flush();
771                     if (channelFuture == null) {
772                         break;
773                     }
774                 } catch (Exception e) {
775                     logger.error("Error sending more queues messages", e);
776                     ctx.fireExceptionCaught(e);
777                 }
778 
779                 if(curTpdu.getParent() != null) {
780                     // Add it to the list of sentButUnacknowledgedTpdus.
781                     // (It seems that the S7 drops the value of the COTP reference id, so we have to use the S7 one)
782                     S7RequestMessage s7RequestMessage = (S7RequestMessage) curTpdu.getParent();
783                     sentButUnacknowledgedTpdus.put(s7RequestMessage.getTpduReference(), curTpdu);
784 
785                     logger.debug("S7 Message with id {} sent", s7RequestMessage.getTpduReference());
786                 }
787                 // TODO: Eventually remove this.
788                 break;
789             } else {
790                 break;
791             }
792         }
793         ctx.flush();
794     }

796     private S7ControllerType lookupControllerType(String articleNumber) {
797         if(!articleNumber.startsWith("6ES7 ")) {
798             return S7ControllerType.S7_ANY;
799         }
800 
801         String model = articleNumber.substring(articleNumber.indexOf(' ') + 1, articleNumber.indexOf(' ') + 2);
802         switch (model) {
803             case "2":
804                 return S7ControllerType.S7_1200;
805             case "5":
806                 return S7ControllerType.S7_1500;
807             case "3":
808                 return S7ControllerType.S7_300;
809             case "4":
810                 return S7ControllerType.S7_400;
811             default:
812                 if(logger.isInfoEnabled()) {
813                     logger.info(String.format("Looking up unknown article number %s", articleNumber));
814                 }
815                 return S7ControllerType.S7_ANY;
816         }
817     }
