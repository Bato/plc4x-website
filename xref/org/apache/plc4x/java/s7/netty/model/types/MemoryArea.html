19  package org.apache.plc4x.java.s7.netty.model.types;
20  
21  import java.util.HashMap;
22  import java.util.Map;
23  
24  /**
25   * (Values determined by evaluating generated ".pcapng" files)
26   */
27  public enum MemoryArea {
28      COUNTERS("C", (byte) 0x1C), /* Renamed from "S7 Counters" */ // TODO: Double check shortName
29      TIMERS("T", (byte) 0x1D), /* Renamed from "S7 Timers" */ // TODO: Double check shortName
30      DIRECT_PERIPHERAL_ACCESS("D", (byte) 0x80), // TODO: Double check shortName
31      INPUTS("I", (byte) 0x81),
32      OUTPUTS("Q", (byte) 0x82),
33      FLAGS("F", (byte) 0x83), // TODO: Double check shortName
34      DATA_BLOCKS("DB", (byte) 0x84),
35      INSTANCE_DATA_BLOCKS("DBI", (byte) 0x85), // TODO: Double check shortName
36      LOCAL_DATA("LD", (byte) 0x86); // TODO: Double check shortName
37  
38      // TODO: I think we should remove these ...
39      /*S7_200_SYSTEM_INFO(null, (byte) 0x03), * Renamed from "System info of 200 family" *
40      S7_200_FLAGS(null, (byte) 0x05), * Renamed from "System flags of 200 family" *
41      S7_200_INPUTS(null, (byte) 0x06), * Renamed from "System inputs of 200 family" *
42      S7_200_OUTPUTS(null, (byte) 0x07), * Renamed from "System outputs of 200 family" *
43      S7_200_IEC_COUNTERS(null, (byte) 0x1E), * Renamed from "IEC counters (200 family)" *
44      S7_200_IEC_TIMERS(null, (byte) 0x1F); * Renamed from "IEC timers (200 family)" */
45  
46      private static final Map<Byte, MemoryArea> map;
47      static {
48          map = new HashMap<>();
49          for (MemoryArea memoryArea : MemoryArea.values()) {
50              map.put(memoryArea.code, memoryArea);
51          }
52      }
53  
54      private final String shortName;
55      private final byte code;
56  
57      MemoryArea(String shortName, byte code) {
58          this.shortName = shortName;
59          this.code = code;
60      }
61  
62      public String getShortName() {
63          return shortName;
64      }
65  
66      public byte getCode() {
67          return code;
68      }
69  
70      public static MemoryArea valueOfShortName(String shortName) {
71          for (MemoryArea value : MemoryArea.values()) {
72              if(value.getShortName().equals(shortName)) {
73                  return value;
74              }
75          }
76          return null;
77      }
78      public static MemoryArea valueOf(byte code) {
79          return map.get(code);
80      }
81  
82  }
