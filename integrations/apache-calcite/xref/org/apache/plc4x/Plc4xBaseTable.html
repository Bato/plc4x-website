47  /**
48   * Base for Stream and "Table" version of the Plc4xTable.
49   * Needs to be subclassed due to usage of "instanceof" in Calcites internals.
50   */
51  public abstract class Plc4xBaseTable extends AbstractTable {
52  
53      private static final Logger logger = LoggerFactory.getLogger(Plc4xBaseTable.class);
54  
55      private final BlockingQueue<Plc4xSchema.Record> queue;
56      private final JobConfiguration conf;
57      private final long tableCutoff;
58      private Plc4xSchema.Record current;
59      private List<String> names;
60  
61      public Plc4xBaseTable(BlockingQueue<Plc4xSchema.Record> queue, JobConfiguration conf, long tableCutoff) {
62          this.tableCutoff = tableCutoff;
63          logger.info("Instantiating new PLC4X Table with configuration: {}", conf);
64          this.queue = queue;
65          this.conf = conf;
66          // Extract names
67          names = new ArrayList<>(conf.getFields().keySet());
68      }
69  
70      @Override
71      public Statistic getStatistic() {
72          return new Statistic() {
73  
74              public Double getRowCount() {
75                  return tableCutoff > 0 ? (double)tableCutoff : null;
76              }
77  
78              public boolean isKey(ImmutableBitSet columns) {
79                  return false;
80              }
81  
82              public List<RelReferentialConstraint> getReferentialConstraints() {
83                  return Collections.emptyList();
84              }
85  
86              public List<RelCollation> getCollations() {
87                  return Collections.singletonList(RelCollationImpl.of(new RelFieldCollation(0, RelFieldCollation.Direction.ASCENDING)));
88              }
89  
90              public RelDistribution getDistribution() {
91                  return RelDistributionTraitDef.INSTANCE.getDefault();
92              }
93          };
94      }
95  
96      @Override
97      public RelDataType getRowType(RelDataTypeFactory typeFactory) {
98          // Create the table spec
99          // Block until the first result is in the queue
100         CompletableFuture<Plc4xSchema.Record> future = CompletableFuture.supplyAsync(new FirstElementFetcher(queue));
101         Plc4xSchema.Record first;
102         try {
103             first = future.get(5, TimeUnit.SECONDS);
104         } catch (InterruptedException e) {
105             Thread.currentThread().interrupt();
106             throw new PlcRuntimeException("Thread was interrupted!", e);
107         } catch (ExecutionException | TimeoutException e) {
108             throw new PlcRuntimeException("Unable to fetch first record and infer arguments!", e);
109         }
110         logger.info("Inferring types for Table '{}' based on values: {}", conf.getName(), first.values);
111         // Extract types
112         List<RelDataType> types = names.stream()
113             .map(n -> {
114                 Object o = first.values.get(n);
115                 logger.debug("Infer field '{}' as class '{}'", n, o.getClass());
116                 return typeFactory.createJavaType(o.getClass());
117             })
118             .collect(Collectors.toList());
119         List<String> pre = new ArrayList<>(Arrays.asList("timestamp", "source"));
120         pre.addAll(names);
121         List<RelDataType> preTypes = Stream.of(Timestamp.class, String.class)
122             .map(typeFactory::createJavaType)
123             .collect(Collectors.toList());
124         preTypes.addAll(types);
125         return typeFactory.createStructType(preTypes, pre);
126     }

128     /**
129      * if tableCutoff is positive, then the row gets limited to that.
130      */
131     public Enumerable<Object[]> scan(DataContext root) {
132         return new AbstractEnumerable<Object[]>() {
133             @Override
134             public Enumerator<Object[]> enumerator() {
135                 return new Enumerator<Object[]>() {
136 
137                     private final AtomicLong counter = new AtomicLong(0);
138 
139                     @Override
140                     public Object[] current() {
141                         List<Object> objects = new ArrayList<>(Arrays.asList(new Timestamp(current.timestamp.toEpochMilli()), current.source));
142                         List<Object> objects2 = names.stream().map(name -> current.values.get(name)).collect(Collectors.toList());
143                         objects.addAll(objects2);
144                         return objects.toArray();
145                     }
146 
147                     @Override
148                     public boolean moveNext() {
149                         try {
150                             current = queue.take();
151                             // If stream, simply return
152                             if (tableCutoff <= 0L) {
153                                 return true;
154                             }
155                             // If table, return if below cutoff
156                             return counter.getAndIncrement() < tableCutoff;
157                         } catch (InterruptedException e) {
158                             Thread.currentThread().interrupt();
159                         }
160                         return false;
161                     }
162 
163                     @Override
164                     public void reset() {
165                         counter.set(0);
166                     }
167 
168                     @Override
169                     public void close() {
170                         // Unimplemented
171                     }
172                 };
173             }
174         };
175     }
176 
177     /**
178      * Waits until a first (non null) element is in the queue
179      */
180     private static class FirstElementFetcher implements Supplier<Plc4xSchema.Record> {
181 
182         private final BlockingQueue<Plc4xSchema.Record> queue;
183 
184         private FirstElementFetcher(BlockingQueue<Plc4xSchema.Record> queue) {
185             this.queue = queue;
186         }
187 
188         @Override
189         public Plc4xSchema.Record get() {
190             Plc4xSchema.Record first;
191             do {
192                 first = queue.peek();
193             } while (first == null);
194             return first;
195         }
196     }
